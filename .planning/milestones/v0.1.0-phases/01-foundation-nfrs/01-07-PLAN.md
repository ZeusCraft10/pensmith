<objective>
Wave 4 — implement the versioned schema migration loader (ARCH-07) plus all 5 zod schemas downstream waves consume.

Per D-37 every persisted JSON file carries a top-level `$schemaVersion: number`. Per D-38 the loader uses zod for runtime validation. Per D-39 the loader REFUSES to downgrade — if disk-version > code-current-version, throw ForwardIncompatError (protects against running older pensmith over newer state files).

Migration chain: each `migrations/{schema}/v{N}_to_v{N+1}.ts` exports a pure function. Loader walks forward from disk-version to current-version, then validates with the current zod schema. Sample migration (state v1→v2) is a no-op identity proving the registry without a real schema change.

The 5 schemas are FOUNDATION-SLICE (D-58, D-59, D-60): they hold only the minimum fields W10/W11 lib code needs. Section state machine (Phase 2), full library entries (Phase 3), full checkpoint snapshots (Phase 5+) are NOT in scope here — those phases extend the schemas via migrations.

**API shape (revision-locked):**

`loadAndMigrate` takes a SINGLE options object (no separate filePath argument). The loader owns reading from disk so call sites stay one-liners:

```ts
const cfg = await loadAndMigrate({
  file: configPath,
  schema: RuntimeConfigSchema,
  schemaName: 'runtime-config',
  currentVersion: CURRENT_RUNTIME_CONFIG_VERSION,
  writeBack: true,
});
```

Each schema module exports a SCHEMA-SPECIFIC current-version constant (e.g. `CURRENT_STATE_VERSION`, not a bare `CURRENT_VERSION`) so a single file can import constants from multiple schema modules without re-aliasing every time.

`runtime-config.providers` is a `z.record(z.string(), ProviderSchema)` keyed by provider id (`'anthropic'`, `'openai'`), not a `z.array`. This is what W11/W13 consumer plans depend on (overlay merge via `{ ...base.providers, ...overlay.providers }` and lookup via `cfg.providers?.[providerId]`).

Output: 1 loader module, 5 schema files, 1 sample migration, 2 test files. ~250 LoC total.

```ts
// bin/lib/migrations/loader.ts — versioned schema loader per ARCH-07.
import { z } from 'zod';
import * as fsp from 'node:fs/promises';
import { atomicWriteFile } from '../atomic-write.ts';

export type Migration = (input: unknown) => unknown;

export interface LoadOptions<TSchema extends z.ZodTypeAny> {
  file: string;
  schema: TSchema;
  schemaName: 'state' | 'library' | 'checkpoint' | 'session-log' | 'runtime-config';
  currentVersion: number;
  migrations?: Record<number, Migration>;
  writeBack?: boolean;        // default false
}

export class ForwardIncompatError extends Error {
  diskVersion: number;
  codeVersion: number;
  constructor(schemaName: string, diskVersion: number, codeVersion: number) {
    super(
      `pensmith: refusing to load ${schemaName} v${diskVersion} (this build supports v${codeVersion}). ` +
      `Upgrade pensmith, or restore from a v${codeVersion}-or-older snapshot.`,
    );
    this.name = 'ForwardIncompatError';
    this.diskVersion = diskVersion;
    this.codeVersion = codeVersion;
  }
}

export class SchemaValidationError extends Error {
  zodIssue: z.ZodIssue[];
  constructor(schemaName: string, issues: z.ZodIssue[]) {
    super(
      `pensmith: ${schemaName} failed schema validation: ` +
      issues.map(i => `${i.path.join('.')}: ${i.message}`).join('; '),
    );
    this.name = 'SchemaValidationError';
    this.zodIssue = issues;
  }
}

function readVersion(raw: unknown): number {
  if (typeof raw !== 'object' || raw === null) return 1;
  const v = (raw as Record<string, unknown>).$schemaVersion;
  if (typeof v === 'number' && Number.isInteger(v) && v >= 1) return v;
  return 1;
}

export async function loadAndMigrate<TSchema extends z.ZodTypeAny>(
  opts: LoadOptions<TSchema>,
): Promise<z.infer<TSchema>> {
  const raw = await fsp.readFile(opts.file, 'utf8');
  let value: unknown = JSON.parse(raw);
  const diskVersion = readVersion(value);

  if (diskVersion > opts.currentVersion) {
    throw new ForwardIncompatError(opts.schemaName, diskVersion, opts.currentVersion);
  }

  let v = diskVersion;
  while (v < opts.currentVersion) {
    const mig = opts.migrations?.[v];
    if (!mig) {
      throw new Error(`pensmith: missing migration ${opts.schemaName} v${v} -> v${v + 1}`);
    }
    value = mig(value);
    v += 1;
  }

  const parsed = opts.schema.safeParse(value);
  if (!parsed.success) {
    throw new SchemaValidationError(opts.schemaName, parsed.error.issues);
  }

  if (v !== diskVersion && opts.writeBack === true) {
    await atomicWriteFile(opts.file, JSON.stringify(parsed.data, null, 2));
  }

  return parsed.data;
}
```

// tests/migrations.test.ts
import test from 'node:test';
import assert from 'node:assert/strict';
import * as fsp from 'node:fs/promises';
import * as path from 'node:path';
import * as os from 'node:os';
import { z } from 'zod';
import { loadAndMigrate, ForwardIncompatError, SchemaValidationError } from '../bin/lib/migrations/loader.ts';
import { Schema as StateSchema, CURRENT_STATE_VERSION } from '../bin/lib/schemas/state.ts';
import v1_to_v2 from '../bin/lib/migrations/state/v1_to_v2.ts';

async function withTmp(fn: (dir: string) => Promise<void>): Promise<void> {
  const dir = await fsp.mkdtemp(path.join(os.tmpdir(), 'pensmith-mig-'));
  try { await fn(dir); } finally { await fsp.rm(dir, { recursive: true, force: true }); }
}

test('forward-incompat throws when diskVersion > currentVersion', async () => {
  await withTmp(async (dir) => {
    const file = path.join(dir, 's.json');
    await fsp.writeFile(file, JSON.stringify({ $schemaVersion: 5, paperId: 'x', createdAt: '2026-05-08T00:00:00.000Z' }));
    let err: unknown;
    try {
      await loadAndMigrate({ file, schema: StateSchema, schemaName: 'state', currentVersion: 1 });
    } catch (e) { err = e; }
    assert.ok(err instanceof ForwardIncompatError, `expected ForwardIncompatError; got ${err}`);
    assert.equal((err as ForwardIncompatError).diskVersion, 5);
    assert.equal((err as ForwardIncompatError).codeVersion, 1);
  });
});

test('missing $schemaVersion is treated as v1', async () => {
  await withTmp(async (dir) => {
    const file = path.join(dir, 's.json');
    // Write a v1 object missing $schemaVersion → readVersion returns 1, then validates
    // against a permissive schema that does NOT require $schemaVersion.
    const PermissiveSchema = z.object({ paperId: z.string(), createdAt: z.string() }).passthrough();
    await fsp.writeFile(file, JSON.stringify({ paperId: 'x', createdAt: '2026-05-08T00:00:00.000Z' }));
    const out = await loadAndMigrate({ file, schema: PermissiveSchema, schemaName: 'state', currentVersion: 1 });
    assert.equal((out as { paperId: string }).paperId, 'x');
  });
});

test('v1 to v2 migration runs and writes back when writeBack:true', async () => {
  await withTmp(async (dir) => {
    const file = path.join(dir, 's.json');
    await fsp.writeFile(file, JSON.stringify({ $schemaVersion: 1, paperId: 'x', createdAt: '2026-05-08T00:00:00.000Z' }));
    const V2Schema = z.object({
      $schemaVersion: z.literal(2),
      paperId: z.string(),
      createdAt: z.string(),
    });
    const out = await loadAndMigrate({
      file,
      schema: V2Schema,
      schemaName: 'state',
      currentVersion: 2,
      migrations: { 1: v1_to_v2 },
      writeBack: true,
    });
    assert.equal(out.$schemaVersion, 2);
    // File on disk is now v2
    const onDisk = JSON.parse(await fsp.readFile(file, 'utf8'));
    assert.equal(onDisk.$schemaVersion, 2);
  });
});

test('writeBack omitted (default false) leaves disk untouched even after migration', async () => {
  await withTmp(async (dir) => {
    const file = path.join(dir, 's.json');
    await fsp.writeFile(file, JSON.stringify({ $schemaVersion: 1, paperId: 'x', createdAt: '2026-05-08T00:00:00.000Z' }));
    const V2Schema = z.object({
      $schemaVersion: z.literal(2),
      paperId: z.string(),
      createdAt: z.string(),
    });
    await loadAndMigrate({
      file,
      schema: V2Schema,
      schemaName: 'state',
      currentVersion: 2,
      migrations: { 1: v1_to_v2 },
      // writeBack omitted → defaults to false
    });
    const onDisk = JSON.parse(await fsp.readFile(file, 'utf8'));
    assert.equal(onDisk.$schemaVersion, 1, 'default writeBack:false should leave disk untouched');
  });
});

test('writeBack:false skips on-disk update', async () => {
