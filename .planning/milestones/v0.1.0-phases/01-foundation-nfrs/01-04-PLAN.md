Per D-15 step 1 — 6 prefix forms to strip (case-insensitive on the prefix, not the body):
- `doi:`, `DOI:`
- `https://doi.org/`, `http://doi.org/`
- `https://dx.doi.org/`, `http://dx.doi.org/`

Per D-15 step 2 — 8 trailing punctuation forms to strip (one pass; do NOT strip recursively):
- `.` `,` `;` `:` `)` `]` `}` `>` `"` `'`
(That's actually 10 chars covering the 8 forms — punctuation/quote/bracket; see D-15 wording)

Per D-15 step 3 — case folding:
- ASCII characters in [A-Z] in the suffix (post-`/`) lowercase
- Non-ASCII bytes (latin-1 supplement, etc.) preserved verbatim
- The PREFIX `10.` stays as-is (already lowercase)

Per D-17 — arXiv formats:
- New: `arXiv:YYMM.NNNNN[vV]` where YY=00-99, MM=01-12, NNNNN=1-5 digit zero-pad
- Old: `subject-class/YYMMNNN` where subject-class is a registry of ~50 known classes
- Strip `arXiv:` / `arxiv:` prefix; lowercase canonical form: `arxiv:2103.00020v2` OR `cs.CL/0301012`

Per D-18 — PMID:
- Accept `12345678` OR `PMID:12345678` (case-insensitive)
- Canonical: bare digits, no prefix

Per D-18 — PMCID:
- Accept `PMC1234567` (case-insensitive)
- Canonical: `PMC1234567` (PMC prefix uppercase)

// bin/lib/doi.ts — DOI/arXiv/PMID/PMCID normalization chokepoint (ARCH-15).
// This is the SOLE file in the repo allowed to use the regex /^10\./
// (Phase 0 D-07 chokepoint, eslint.config.js exemption is in place).
//
// Spec (D-15, D-17, D-18):
//   normalizeDoi:
//     1. trim whitespace
//     2. strip prefix (one of 6 forms, case-insensitive on prefix)
//     3. strip trailing punctuation (one pass: . , ; : ) ] } > " ')
//     4. lowercase ASCII alpha in suffix, preserve non-ASCII bytes verbatim
//     5. validate: /^10\.\d{4,9}\/[^\s]+$/  → return string OR null
//
//   normalizeArxiv:
//     - strip arxiv:/arXiv:/ARXIV: prefix
//     - validate new format (YYMM.NNNNN[vV]) OR old format (class/YYMMNNN)
//     - canonical: 'arxiv:' + body for new; 'class/YYMMNNN' (no prefix) for old
//
//   normalizePmid:
//     - accept digits-only OR PMID:digits (case-insensitive prefix)
//     - canonical: bare digits
//
//   normalizePmcid:
//     - accept PMC-digits (case-insensitive prefix)
//     - canonical: 'PMC' + digits

const DOI_PREFIXES = [
  'https://doi.org/',
  'http://doi.org/',
  'https://dx.doi.org/',
  'http://dx.doi.org/',
  'doi:',
  'DOI:',
];

const TRAILING_PUNCT = /[.,;:)\]}>"']+$/;

const DOI_VALID = /^10\.\d{4,9}\/\S+$/;

function lowerAsciiOnly(s: string): string {
  // Lowercase only [A-Z], preserve everything else verbatim.
  return s.replace(/[A-Z]/g, (c) => c.toLowerCase());
}

export function normalizeDoi(input: string): string | null {
  if (typeof input !== 'string') return null;
  let s = input.trim();
  if (!s) return null;
  // Strip prefix (case-insensitive match)
  for (const p of DOI_PREFIXES) {
    if (s.toLowerCase().startsWith(p.toLowerCase())) {
      s = s.slice(p.length);
      break;
    }
  }
  // Strip trailing punctuation in one pass
  s = s.replace(TRAILING_PUNCT, '');
  if (!s) return null;
  // Split prefix/suffix on first '/'
  const idx = s.indexOf('/');
  if (idx < 0) return null;
  const prefix = s.slice(0, idx);
  const suffix = s.slice(idx + 1);
  // Lowercase ASCII letters in BOTH halves; non-ASCII preserved
  const canonical = lowerAsciiOnly(prefix) + '/' + lowerAsciiOnly(suffix);
  return DOI_VALID.test(canonical) ? canonical : null;
}

export function isDoi(s: string): boolean {
  return normalizeDoi(s) !== null;
}

const ARXIV_NEW = /^(?:arxiv:)?(\d{4}\.\d{4,5}(?:v\d+)?)$/i;
// Old format subject classes (D-17 — registry of major archives).
const ARXIV_OLD_CLASSES = new Set([
  'cs', 'math', 'physics', 'astro-ph', 'cond-mat', 'gr-qc', 'hep-ex',
  'hep-lat', 'hep-ph', 'hep-th', 'math-ph', 'nlin', 'nucl-ex', 'nucl-th',
  'q-bio', 'quant-ph', 'stat',
  // Common dotted subclasses
  'cs.CL', 'cs.AI', 'cs.LG', 'cs.CV', 'cs.IR', 'cs.NE', 'cs.DS', 'cs.SE',
  'math.AG', 'math.AT', 'math.CA', 'math.CO', 'math.NT', 'math.PR',
  'astro-ph.CO', 'astro-ph.EP', 'astro-ph.GA', 'astro-ph.HE', 'astro-ph.IM', 'astro-ph.SR',
]);

function isKnownArxivClass(c: string): boolean {
  return ARXIV_OLD_CLASSES.has(c);
}

export function normalizeArxiv(input: string): string | null {
  if (typeof input !== 'string') return null;
  const s = input.trim();
  if (!s) return null;
  // New format
  const newMatch = ARXIV_NEW.exec(s);
  if (newMatch) {
    return 'arxiv:' + newMatch[1].toLowerCase();
  }
  // Old format: optional arxiv: prefix + class/YYMMNNN
  const stripped = s.replace(/^arxiv:/i, '');
  const slashIdx = stripped.indexOf('/');
  if (slashIdx < 0) return null;
  const cls = stripped.slice(0, slashIdx);
  const body = stripped.slice(slashIdx + 1);
  if (!isKnownArxivClass(cls)) return null;
  if (!/^\d{7}$/.test(body)) return null;
  return `${cls}/${body}`;
}

export function isArxiv(s: string): boolean {
  return normalizeArxiv(s) !== null;
}

export function normalizePmid(input: string): string | null {
  if (typeof input !== 'string') return null;
  const s = input.trim().replace(/^pmid:/i, '');
  return /^\d+$/.test(s) && s.length <= 9 ? s : null;
}

export function isPmid(s: string): boolean {
  return normalizePmid(s) !== null;
}

export function normalizePmcid(input: string): string | null {
  if (typeof input !== 'string') return null;
  const m = /^pmc(\d+)$/i.exec(input.trim());
  return m ? `PMC${m[1]}` : null;
}

export function isPmcid(s: string): boolean {
  return normalizePmcid(s) !== null;
}

// tests/doi.test.ts — spec tests for bin/lib/doi.ts
import test from 'node:test';
import assert from 'node:assert/strict';
import {
  normalizeDoi, normalizeArxiv, normalizePmid, normalizePmcid,
  isDoi, isArxiv, isPmid, isPmcid,
} from '../bin/lib/doi.ts';

test('normalizeDoi: bare DOI passes through canonical', () => {
  assert.equal(normalizeDoi('10.1038/s41586-021-03819-2'), '10.1038/s41586-021-03819-2');
});

test('normalizeDoi: all 6 prefix forms strip', () => {
  const target = '10.1038/s41586-021-03819-2';
  assert.equal(normalizeDoi(`doi:${target}`), target);
  assert.equal(normalizeDoi(`DOI:${target}`), target);
  assert.equal(normalizeDoi(`https://doi.org/${target}`), target);
  assert.equal(normalizeDoi(`http://doi.org/${target}`), target);
  assert.equal(normalizeDoi(`https://dx.doi.org/${target}`), target);
  assert.equal(normalizeDoi(`http://dx.doi.org/${target}`), target);
});

test('normalizeDoi: all 8 trailing punctuation forms strip', () => {
  const target = '10.1038/s41586-021-03819-2';
  for (const p of ['.', ',', ';', ':', ')', ']', '}', '>', '"', "'"]) {
    assert.equal(normalizeDoi(target + p), target, `trailing ${p} should strip`);
  }
});

test('normalizeDoi: lowercases ASCII suffix', () => {
  assert.equal(normalizeDoi('10.1038/S41586-021-03819-2'), '10.1038/s41586-021-03819-2');
  assert.equal(normalizeDoi('10.1038/AAA'), '10.1038/aaa');
});

test('normalizeDoi: preserves non-ASCII bytes verbatim', () => {
  assert.equal(normalizeDoi('10.1234/résumé'), '10.1234/résumé');
  assert.equal(normalizeDoi('10.1234/abç'), '10.1234/abç');
});

test('normalizeDoi: trims whitespace', () => {
  assert.equal(normalizeDoi('  10.1038/s41586  '), '10.1038/s41586');
});

test('normalizeDoi: garbage returns null', () => {
  assert.equal(normalizeDoi(''), null);
  assert.equal(normalizeDoi('not-a-doi'), null);
  assert.equal(normalizeDoi('10.'), null);
  assert.equal(normalizeDoi('10.123'), null);    // missing /
  assert.equal(normalizeDoi('10.1/x'), null);     // 4-digit registrant requirement
  assert.equal(normalizeDoi('foo/bar'), null);
});

test('normalizeArxiv: new format YYMM.NNNNN[vV]', () => {
  assert.equal(normalizeArxiv('arXiv:2103.00020'), 'arxiv:2103.00020');
  assert.equal(normalizeArxiv('arXiv:2103.00020v2'), 'arxiv:2103.00020v2');
  assert.equal(normalizeArxiv('2103.00020'), 'arxiv:2103.00020');
});

test('normalizeArxiv: old format subject-class/YYMMNNN', () => {
  assert.equal(normalizeArxiv('cs.CL/0301012'), 'cs.CL/0301012');
  assert.equal(normalizeArxiv('astro-ph/0301012'), 'astro-ph/0301012');
  assert.equal(normalizeArxiv('arxiv:cs.CL/0301012'), 'cs.CL/0301012');
});

test('normalizeArxiv: garbage returns null', () => {
  assert.equal(normalizeArxiv(''), null);
  assert.equal(normalizeArxiv('garbage'), null);
  assert.equal(normalizeArxiv('cs.UNKNOWN/0301012'), null);
});

test('normalizePmid: bare digits + PMID: prefix', () => {
  assert.equal(normalizePmid('12345678'), '12345678');
  assert.equal(normalizePmid('PMID:12345678'), '12345678');
  assert.equal(normalizePmid('pmid:12345678'), '12345678');
  assert.equal(normalizePmid(' 12345678 '), '12345678');
});

test('normalizePmid: garbage returns null', () => {
  assert.equal(normalizePmid(''), null);
  assert.equal(normalizePmid('abc'), null);
  assert.equal(normalizePmid('1234567890'), null); // > 9 digits
});

test('normalizePmcid: PMC + digits', () => {
  assert.equal(normalizePmcid('PMC1234567'), 'PMC1234567');
  assert.equal(normalizePmcid('pmc1234567'), 'PMC1234567');
});

test('normalizePmcid: missing prefix → null', () => {
  assert.equal(normalizePmcid('1234567'), null);
});

test('isDoi / isArxiv / isPmid / isPmcid mirror normalize* != null', () => {
  assert.equal(isDoi('10.1038/x'), true);
  assert.equal(isDoi('xyz'), false);
  assert.equal(isArxiv('arXiv:2103.00020'), true);
  assert.equal(isArxiv('xyz'), false);
  assert.equal(isPmid('12345678'), true);
  assert.equal(isPmid('xyz'), false);
  assert.equal(isPmcid('PMC1234567'), true);
  assert.equal(isPmcid('xyz'), false);
});
