<objective>
Land the 7 source adapters that the research workflow will consume. Each one is a thin wrapper: HTTP call → parse → return SourceCandidate. All responses are recorded once as cassettes (committed JSON ≤50KB each) and replayed offline in PR-time CI. Live re-recording happens only via the weekly cron (Plan 09).

Purpose: Until adapters exist, the research workflow body (Plan 06) cannot run. Until cassettes exist, PR-time CI cannot prove adapter correctness offline.

Additionally, this plan lands `bin/lib/bibtex-write.ts` — the SourceCandidate[] -> BibTeX serializer that workflows/research.md (Plan 06) invokes to emit `.paper/CITATIONS.bib` (D-20 LOCKED canonical path). It rides the D-19 LOCKED citation-js chokepoint and the D-07 LOCKED atomic-write chokepoint. Without this helper, the research verb cannot produce the canonical bib that the verify verb (Plan 06 verify.md body) reads via bin/lib/citations.ts at Pass-1 time, and the Plan 09 tier-contract research case has no .paper/CITATIONS.bib to assert against.

Output: 7 adapter modules + 1 registry index + 1 cassette loader + 8 cassette JSON files + 1 BibTeX serializer + 1 serializer test.
</objective>

       ```typescript
       import nock from 'nock';
       import { readdirSync, readFileSync, existsSync } from 'node:fs';
       import { join } from 'node:path';

       const CASSETTES_ROOT = 'tests/fixtures/cassettes';

       export function isOfflineMode(): boolean {
         return process.env.PENSMITH_NETWORK_TESTS !== '1';
       }

       interface Cassette {
         scope: string;
         method: 'GET' | 'POST';
         path: string;
         status: number;
         response: unknown;
         responseHeaders?: Record<string, string>;
       }

       export function loadCassettes(adapter: string): void {
         if (!isOfflineMode()) return;
         const dir = join(CASSETTES_ROOT, adapter);
         if (!existsSync(dir)) {
           throw new Error(`No cassette directory for adapter "${adapter}" at ${dir} (D-23)`);
         }
         const files = readdirSync(dir).filter((f) => f.endsWith('.json'));
         for (const f of files) {
           const cassettes: Cassette[] = JSON.parse(readFileSync(join(dir, f), 'utf8'));
           for (const c of cassettes) {
             nock(c.scope)
               .intercept(c.path, c.method)
               .reply(c.status, c.response, c.responseHeaders ?? {});
           }
         }
         // Lockdown: any unmatched request throws
         nock.disableNetConnect();
       }

       export function clearCassettes(): void {
         nock.cleanAll();
         nock.enableNetConnect();
       }
       ```

       import { Cite } from './citations.js';          // D-19 LOCKED chokepoint — NEVER import 'citation-js' directly
       import { atomicWriteFile } from './atomic-write.js';  // D-07 LOCKED chokepoint — NEVER call fs.writeFile
       import type { SourceCandidate } from './schemas/source-candidate.js';

       interface CslAuthor { family: string; given?: string }
       interface CslEntry {
         type: 'article-journal' | 'paper-conference' | 'article' | 'book';
         title: string;
         author: CslAuthor[];
         issued?: { 'date-parts': [[number]] };
         DOI?: string;
         ISBN?: string;
         number?: string;  // arXiv id surfaces here per CSL-JSON convention for arxiv preprints
       }

       function toCsl(c: SourceCandidate): CslEntry | null {
         const hasId = Boolean(c.doi) || Boolean((c as { isbn?: string }).isbn) || Boolean((c as { arxivId?: string }).arxivId);
         if (!hasId) return null;
         return {
           type: c.source === 'arxiv' ? 'article' : 'article-journal',
           title: c.title,
           author: (c.authors ?? []).map(a => ({ family: a.family, given: a.given })),
           issued: c.year ? { 'date-parts': [[c.year]] } : undefined,
           DOI: c.doi ?? undefined,
           ISBN: (c as { isbn?: string }).isbn ?? undefined,
           number: (c as { arxivId?: string }).arxivId ?? undefined,
         };
       }

       export async function writeBibtex(candidates: SourceCandidate[], targetPath: string): Promise<void> {
         const cslEntries = candidates.map(toCsl).filter((e): e is CslEntry => e !== null);
         // Empty array is valid — Cite([]) returns an empty bib; we still write the file so verify.md does not ENOENT.
         const cite = new Cite(cslEntries);
         const bibtex = cite.format('bibtex', { format: 'text' }) as string;
         // Sort entries by citekey for git-diff stability.
         const blocks = bibtex.split(/\n(?=@)/).map(b => b.trim()).filter(Boolean);
         blocks.sort((a, b) => {
           const ka = a.match(/^@\w+\{([^,]+),/)?.[1] ?? '';
           const kb = b.match(/^@\w+\{([^,]+),/)?.[1] ?? '';
           return ka.localeCompare(kb);
         });
         await atomicWriteFile(targetPath, blocks.join('\n\n') + (blocks.length ? '\n' : ''));
       }
