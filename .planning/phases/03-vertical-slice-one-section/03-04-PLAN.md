       ```typescript
       import nock from 'nock';
       import { readdirSync, readFileSync, existsSync } from 'node:fs';
       import { join } from 'node:path';

       const CASSETTES_ROOT = 'tests/fixtures/cassettes';

       export function isOfflineMode(): boolean {
         return process.env.PENSMITH_NETWORK_TESTS !== '1';
       }

       interface Cassette {
         scope: string;
         method: 'GET' | 'POST';
         path: string;
         status: number;
         response: unknown;
         responseHeaders?: Record<string, string>;
       }

       export function loadCassettes(adapter: string): void {
         if (!isOfflineMode()) return;
         const dir = join(CASSETTES_ROOT, adapter);
         if (!existsSync(dir)) {
           throw new Error(`No cassette directory for adapter "${adapter}" at ${dir} (D-23)`);
         }
         const files = readdirSync(dir).filter((f) => f.endsWith('.json'));
         for (const f of files) {
           const cassettes: Cassette[] = JSON.parse(readFileSync(join(dir, f), 'utf8'));
           for (const c of cassettes) {
             nock(c.scope)
               .intercept(c.path, c.method)
               .reply(c.status, c.response, c.responseHeaders ?? {});
           }
         }
         // Lockdown: any unmatched request throws
         nock.disableNetConnect();
       }

       export function clearCassettes(): void {
         nock.cleanAll();
         nock.enableNetConnect();
       }
       ```

<threat_model>

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| External API → adapter parse | Untrusted JSON/XML; adapters must not crash on missing fields, must return SourceCandidate[] only after zod-validated mapping. |
| process.env.PENSMITH_S2_API_KEY → outbound HTTP header | Value flows from env to header bytes; must NEVER be logged, persisted, or returned in adapter output. |
| Cassette JSON → nock | Cassettes are committed source; their content is the "trusted" simulation of upstream APIs. Hash-pinning is OUT of scope here (overkill); diff review in PR is the gate. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-3-13 | Spoofing | Adapter response shape drift breaks parser silently | mitigate | Per-adapter unit test against committed cassette; weekly cron re-records and auto-opens PR (Plan 09); cassette diff in PR is human review point. |
| T-3-02 | Information Disclosure | PENSMITH_S2_API_KEY value persisted via adapter logging | mitigate | semanticscholar.ts uses the value ONLY in outbound headers; no console.log/info; tests/runtime.test.ts asserts capabilities serialization doesn't contain secret. |
| T-3-NET-01 | Information Disclosure | Live HTTP from PR-time CI leaks IP/usage to external services | mitigate | http-mock.loadCassettes() calls nock.disableNetConnect() in offline mode; PENSMITH_NETWORK_TESTS=0 default; CI never sets =1. |
| T-3-XML-01 | Denial of Service | Malicious XML in arxiv response (XXE / billion-laughs) | mitigate (low) | Regex-based extractor (no XML parser, no entity expansion). RESEARCH.md notes this trade-off explicitly. |
| T-3-DOS-03 | Denial of Service | Huge adapter response in cassette | mitigate | cassette-size.test.ts enforces ≤51200 bytes per cassette file; PR fails if any cassette exceeds. |

</threat_model>
