    Create `bin/lib/author-normalize.ts`:
    ```typescript
    import { nfkcNormalize } from './normalize.js';

    const PARTICLES = new Set([
      'van', 'von', 'de', 'den', 'der', 'la', 'le', 'di', 'da', 'del',
      'dos', 'do', 'du', 'el', 'al', 'bin', 'ben', 'mac', 'mc', "o'", 'ter',
    ]);

    /**
     * Extracts the surname/family name from a single author string for D-11 Pass-1 AND-gate.
     * Returns lowercased, NFKC-normalized, diacritic-stripped surname (or full multi-word
     * surname for particled names like "van den Berg").
     *
     * @example
     * firstAuthorSurname("Vaswani, A.") // → "vaswani"
     * firstAuthorSurname("R. van den Berg") // → "van den berg"
     * firstAuthorSurname("Müller-Schmidt") // → "muller-schmidt"
     */
    export function firstAuthorSurname(authorString: string): string {
      const normalized = nfkcNormalize(authorString).toLowerCase().trim();
      if (!normalized) return '';

      // Comma-separated → surname-first ("Vaswani, A." or "van den Berg, R.")
      if (normalized.includes(',')) {
        const before = normalized.split(',')[0].trim();
        return before;
      }

      // Space-separated: detect particle run + surname tail
      const tokens = normalized.split(/\s+/).filter(Boolean);
      if (tokens.length === 0) return '';
      if (tokens.length === 1) return tokens[0];

      // Walk backwards: find the run of particles followed by surname tokens
      // The surname is the last token (or sequence) NOT consisting of pure initials
      // Initials look like "a." or "a" (single letter, possibly with dot)
      const isInitial = (t: string) => /^[a-z]\.?$/.test(t);

      // Find first non-initial token from the right; the surname starts there
      let surnameStart = tokens.length - 1;
      while (surnameStart > 0 && isInitial(tokens[surnameStart])) surnameStart--;

      // Extend backwards through particles
      while (surnameStart > 0 && PARTICLES.has(tokens[surnameStart - 1])) surnameStart--;

      return tokens.slice(surnameStart).join(' ');
    }

    /**
     * Batch-normalize an author list to surnames. Used by bin/lib/verify/pass1.ts.
     */
    export function normalizeAuthorList(authors: string[]): string[] {
      return authors.map(firstAuthorSurname);
    }
    ```
