       ```typescript
       };

       export function loadPrompt(name: string): string {
         const path = `templates/prompts/${name}.md`;
         const bytes = readFileSync(path);
         const actual = createHash('sha256').update(bytes).digest('hex');
         const expected = EXPECTED_PROMPT_HASHES[name];
         if (!expected) throw new Error(`Unknown prompt "${name}" — no expected hash registered`);
         if (actual !== expected) {
           throw new Error(`Prompt "${name}" drifted at runtime. Expected ${expected}, got ${actual}. Update EXPECTED_PROMPT_HASHES in prompt-loader.ts (single source of truth — tests/repo-files.test.ts imports this map per WN-3) together (D-12). Note: pass1-fuzzy-judge + pass3-quote-checker are D-13 DORMANT in Phase 3 — if you are seeing this error for one of those slugs at runtime in Phase 3, the workflow body is incorrectly invoking a dormant prompt.`);
         }
         // Strip YAML frontmatter
         const text = bytes.toString('utf8');
         const body = text.startsWith('---') ? text.split(/^---\s*$/m).slice(2).join('---').trimStart() : text;
         return body;
       }

       export function interpolate(template: string, vars: Record<string, string>): string {
         return template.replace(/\{\{(\w+)\}\}/g, (_, key) => {
           if (!(key in vars)) throw new Error(`interpolate: missing var "${key}"`);
           return vars[key];
         });
       }
       ```

    ```typescript
    /**
     * Deterministic Pass-1 citation-integrity check. NO LLM (D-13 LOCKED).
     * - FABRICATED if citekey absent from .paper/CITATIONS.bib OR DOI doesn't resolve via Crossref
     * - MIS-CITED if title/author JW < thresholds OR retracted flag set OR multi-DOI redirect mismatch OR empty-field
     * - OK otherwise
     *
     * CYCLE-2 H-4 REVIEWS CONVERGENCE — Pass1 signature lock:
     * THE canonical signature is `runPass1(draftMd: string, citationsBibPath: string)`.
     * For unit-level fixture testing (where the caller already has a parsed claimed citation
     * shape, not a DRAFT.md + bib path), export `runPass1Unit(input)` below that operates on
     * a single fixture entry. Plan 09's tests/known-bad-citations.test.ts calls runPass1Unit,
     * NOT runPass1, so the fixture loop can iterate without rebuilding draft + bib for each
     * case. Both functions share the same per-citekey verdict logic via a private helper.
     */
    export async function runPass1(draftMd: string, citationsBibPath: string): Promise<Pass1Result[]> {
      const bib = parseBibtex(readFileSync(citationsBibPath, 'utf8'));
      const bibByCitekey = new Map(bib.map((e: any) => [e.id, e]));
      const citekeys = [...draftMd.matchAll(/\[@([a-z][a-z0-9_-]*)\]/g)].map(m => m[1]);
      const unique = [...new Set(citekeys)];
      const results: Pass1Result[] = [];
      for (const ck of unique) {
        const claimed = bibByCitekey.get(ck);
        if (!claimed) {
          results.push({ citekey: ck, verdict: 'FABRICATED', titleJW: 0, authorJW: 0, reason: `citekey not in .paper/CITATIONS.bib (drafter invented)` });
          continue;
        }
        // CYCLE-2 H-2 REVIEWS CONVERGENCE (D-14 author shape lock):
        // Normalize claimed (citation-js BibTeX shape) into D-14 string[] form ONCE at the boundary
        // so the comparison below operates on a uniform shape. The internal canonical type used by
        // pass1.ts comparison is D-14 `string[]` ("Family, Given"); BibTeX-only shape stays scoped
        // to this normalize step. NEVER read `claimed.author[0].family` downstream.
        const claimedAuthorsD14: string[] = ((claimed.author as Array<{family?: string; given?: string}>) ?? [])
          .map(a => {
            const family = String(a?.family ?? '').trim();
            const given = String(a?.given ?? '').trim();
            return given ? `${family}, ${given}` : family;
          })
          .filter(Boolean);
        // Field-presence sub-gate (REVIEWS amendment OpenCode MEDIUM #5)
        if (!claimed.title || (Array.isArray(claimed.title) ? !claimed.title[0] : !claimed.title) || claimedAuthorsD14.length === 0) {
          results.push({ citekey: ck, verdict: 'MIS-CITED', titleJW: 0, authorJW: 0, reason: 'claimed citation metadata incomplete (empty title or no authors)' });
          continue;
        }
        if (claimed.retracted) {
          results.push({ citekey: ck, verdict: 'MIS-CITED', titleJW: 0, authorJW: 0, reason: 'cited a retracted work (per Retraction Watch cross-check at research time)' });
          continue;
        }
        if (!claimed.DOI) {
          results.push({ citekey: ck, verdict: 'FABRICATED', titleJW: 0, authorJW: 0, reason: 'no DOI in citation entry (cannot verify)' });
          continue;
        }
        const actual = await sources.crossref.fetchById(claimed.DOI);
        if (!actual) {
          results.push({ citekey: ck, verdict: 'FABRICATED', titleJW: 0, authorJW: 0, reason: `DOI ${claimed.DOI} did not resolve via Crossref` });
          continue;
        }
        // Both sides now D-14 string[] (actual.authors from SourceCandidate; claimedAuthorsD14 from normalize step above).
        const titleJW = jaroWinkler(actual.title, Array.isArray(claimed.title) ? claimed.title[0] : claimed.title);
        const authorJW = jaroWinkler(
          firstAuthorSurname(actual.authors?.[0] ?? ''),
          firstAuthorSurname(claimedAuthorsD14[0] ?? '')
        );
        // Multi-DOI redirect handling
        const actualDoi = (actual as any).doi ?? claimed.DOI;
        if (actualDoi !== claimed.DOI) {
          if (titleJW >= 0.98 && authorJW >= 0.95) {
            results.push({ citekey: ck, verdict: 'OK', titleJW, authorJW, reason: `multi-DOI redirect: ${claimed.DOI} → ${actualDoi}, strict-match OK` });
          } else {
            results.push({ citekey: ck, verdict: 'MIS-CITED', titleJW, authorJW, reason: `claimed DOI ${claimed.DOI} resolves to different work (canonical: ${actualDoi})` });
          }
          continue;
        }
        if (titleJW >= TITLE_JW_THRESHOLD && authorJW >= AUTHOR_JW_THRESHOLD) {
          results.push({ citekey: ck, verdict: 'OK', titleJW, authorJW, reason: 'D-11 AND-gate passed' });
        } else {
          results.push({ citekey: ck, verdict: 'MIS-CITED', titleJW, authorJW, reason: `JW below threshold (title=${titleJW.toFixed(2)}/${TITLE_JW_THRESHOLD}, author=${authorJW.toFixed(2)}/${AUTHOR_JW_THRESHOLD})` });
        }
      }
      return results;
    }
    ```

    ```typescript
          continue;
        }
        const u = await sources.unpaywall.fetchById(claimed.DOI);
        const oaUrl = u?.oa_pdf_url;
        if (!oaUrl) {
          results.push({ citekey: q.citekey, quoteSnippet: q.text.slice(0, 40), verdict: 'PDF_UNAVAILABLE', levRatio: 0, reason: `No OA PDF available for DOI ${claimed.DOI}` });
          continue;

        const buf = Buffer.isBuffer(resp.body) ? resp.body : Buffer.from(resp.body as string);
        const text = await extractPdfText(buf);
        if (text.replace(/\s/g, '').length < 50) {
          results.push({ citekey: q.citekey, quoteSnippet: q.text.slice(0, 40), verdict: 'TEXT_UNAVAILABLE', levRatio: 0, reason: 'PDF appears image-only or scanned (<50 non-whitespace chars)' });
          continue;
        }
        const ratio = levenshteinSubstring(nfkcNormalize(q.text), nfkcNormalize(text));
        if (ratio >= QUOTE_LEV_THRESHOLD) {
          results.push({ citekey: q.citekey, quoteSnippet: q.text.slice(0, 40), verdict: 'OK', levRatio: ratio, reason: 'levenshtein-substring above threshold' });

          results.push({ citekey: q.citekey, quoteSnippet: q.text.slice(0, 40), verdict: 'NOT_FOUND', levRatio: ratio, reason: `quote not found in OA PDF (lev=${ratio.toFixed(3)} < ${QUOTE_LEV_THRESHOLD})` });
        }
      }
    ```

    ```typescript
    /**
     * Quote extractor per REVIEWS amendment (Codex HIGH #4, OpenCode HIGH #4).
     * Rules:
     *   - Block quotes (lines starting with '> '): include if word count >= 10
     *   - Inline quotes (text wrapped in "..." / "..." / '...'): include if >= 10 words AND >= 60 chars
     *   - Multi-paragraph block quote (consecutive '> ' lines): treat as one quote, summed word count
     *   - Strip Pandoc tokens [@citekey] BEFORE counting words
     *   - Quote MUST be associated with the immediately-following [@citekey] (within 200 chars)
     */
    export interface ExtractedQuote {
      text: string;
      citekey: string;
      kind: 'block' | 'inline';
    }

    const MIN_WORDS = 10;
    const MIN_INLINE_CHARS = 60;

    export function extractQuotes(draftMd: string): ExtractedQuote[] {
      const stripCites = (s: string) => s.replace(/\[@[a-z][a-z0-9_-]*\]/g, '').replace(/\s+/g, ' ').trim();
      const wordCount = (s: string) => stripCites(s).split(/\s+/).filter(Boolean).length;
      const out: ExtractedQuote[] = [];

      // Block quotes: find runs of '> ' lines
      const lines = draftMd.split('\n');
      let blockBuf: string[] = [];
      let blockEndIdx = -1;
      for (let i = 0; i <= lines.length; i++) {
        const isBlock = i < lines.length && lines[i].startsWith('> ');
        if (isBlock) {
          blockBuf.push(lines[i].slice(2));
          blockEndIdx = i;
        } else if (blockBuf.length > 0) {
          const text = blockBuf.join(' ');
          if (wordCount(text) >= MIN_WORDS) {
            // Find first [@citekey] in next 200 chars after blockEndIdx line
            const after = lines.slice(blockEndIdx + 1, blockEndIdx + 5).join(' ');
            const m = after.match(/\[@([a-z][a-z0-9_-]*)\]/);
            if (m) out.push({ text: stripCites(text), citekey: m[1], kind: 'block' });
          }
          blockBuf = [];
        }
      }

      // Inline quotes
      const inlineRe = /["“]([^"”]{60,})["”]\s*\[@([a-z][a-z0-9_-]*)\]/g;
      for (const m of draftMd.matchAll(inlineRe)) {
        if (wordCount(m[1]) >= MIN_WORDS) {
          out.push({ text: stripCites(m[1]), citekey: m[2], kind: 'inline' });
        }
      }
      return out;
    }
    ```
