       ```typescript
       import { writeFile, rename, mkdir } from 'node:fs/promises';
       import { dirname } from 'node:path';
       import { lock } from 'proper-lockfile';
       import { HandoffSchema, type Handoff, HANDOFF_MAX_BYTES } from './schemas/handoff.js';

       export const HANDOFF_PATH = '.paper/HANDOFF.json';

       export function assembleHandoff(input: {
         phase: string;
         wave: number;
         resumePrompt: string;
         pointers: Handoff['pointers'];
       }): Handoff {
         const candidate = {
           schemaVersion: '1' as const,
           phase: input.phase,
           wave: input.wave,
           resumePrompt: input.resumePrompt.slice(0, 300),  // D-12 invariant: the cap is mechanical (in this code), NOT delegated to a prompt. There is no draft-summary prompt in the D-12 LOCKED set; resumePrompt is a deterministic template literal constructed by the caller (e.g. pre-compact.ts) and truncated here.
           pointers: input.pointers.slice(0, 20),
         };
         return HandoffSchema.parse(candidate);  // throws if size > 5120 via refinement
       }

       export async function writeHandoff(handoff: Handoff): Promise<void> {
         // Schema-validate (defense-in-depth — caller should already have done this)
         HandoffSchema.parse(handoff);
         const content = JSON.stringify(handoff, null, 2);
         if (Buffer.byteLength(content, 'utf8') > HANDOFF_MAX_BYTES) {
           throw new Error(`HANDOFF serialized size ${Buffer.byteLength(content, 'utf8')} exceeds ${HANDOFF_MAX_BYTES} bytes (D-17)`);
         }
         await mkdir(dirname(HANDOFF_PATH), { recursive: true });
         const release = await lock(HANDOFF_PATH, {
           retries: { retries: 5, minTimeout: 50 },
           realpath: false,  // HANDOFF.json may not exist yet
         });
         try {
           await writeFile(`${HANDOFF_PATH}.tmp`, content, 'utf8');
           await rename(`${HANDOFF_PATH}.tmp`, HANDOFF_PATH);
         } finally {
           await release();
         }
       }
       ```

    2. Create `bin/lib/frontmatter.ts`:

       ```typescript
       import { parseDocument, type Document } from 'yaml';

       const FRONTMATTER_RE = /^---\r?\n([\s\S]*?)\r?\n---\r?\n([\s\S]*)$/;

       export function parseFrontmatter(text: string): { frontmatter: Record<string, unknown>; body: string } {
         const match = FRONTMATTER_RE.exec(text);
         if (!match) return { frontmatter: {}, body: text };
         const doc = parseDocument(match[1]);
         return { frontmatter: doc.toJSON() ?? {}, body: match[2] };
       }

       export function serializeFrontmatter(frontmatter: Record<string, unknown>): string {
         const doc = parseDocument('');
         for (const [k, v] of Object.entries(frontmatter)) {
           doc.set(k, v);
         }
         return `---\n${doc.toString()}---\n`;
       }

       /**
        * Round-trip safe: preserves comments + key order in the existing frontmatter
        * by using the yaml Document API (RESEARCH.md pitfall #7).
        */
       export function updateFrontmatter(
         text: string,
         mutator: (fm: Record<string, unknown>) => void,
       ): string {
         const match = FRONTMATTER_RE.exec(text);
         if (!match) {
           // No existing frontmatter — fall back to fresh serialize
           const fm: Record<string, unknown> = {};
           mutator(fm);
           return serializeFrontmatter(fm) + text;
         }
         const doc = parseDocument(match[1]);
         const fm = doc.toJSON() ?? {};
         mutator(fm);
         // Re-apply mutated values onto the Document to preserve comments
         for (const [k, v] of Object.entries(fm)) {
           doc.set(k, v);
         }
         return `---\n${doc.toString()}---\n${match[2]}`;
       }
       ```

       ```typescript
       export default async function preCompactHook(input: unknown): Promise<void> {
         try {
           // 1. Read STATE.md to extract phase + wave
           const stateText = existsSync('.planning/STATE.md')
             ? readFileSync('.planning/STATE.md', 'utf8')
             : '';
           const phaseMatch = /currentPhaseId:\s*([^\s\n]+)/.exec(stateText);
           const waveMatch = /currentWave:\s*(\d+)/.exec(stateText);
           const phase = phaseMatch?.[1] ?? 'unknown';
           const wave = waveMatch ? Number(waveMatch[1]) : 0;

           // 2. Find current section (latest mtime under .paper/sections/)
           const pointers: { kind: 'plan'|'draft'|'verification'|'research'|'state'; path: string }[] = [];
           if (existsSync('.planning/STATE.md')) {

           }
           if (existsSync('.paper/sections')) {
             const sectionDirs = readdirSync('.paper/sections').map(d => `.paper/sections/${d}`).filter(d => statSync(d).isDirectory());
             sectionDirs.sort((a, b) => statSync(b).mtimeMs - statSync(a).mtimeMs);
             const latest = sectionDirs[0];

               for (const [file, kind] of [['PLAN.md', 'plan'], ['DRAFT.md', 'draft'], ['VERIFICATION.md', 'verification'], ['RESEARCH.md', 'research']] as const) {
       ```

       ```typescript
       import { dirname } from 'node:path';

       const CHECKPOINTS_PATH = '.claude/CHECKPOINTS.jsonl';
       const THROTTLE_MS = 60_000;

       export default async function postToolUseHook(input: { tool?: string }): Promise<void> {
         try {
           // Read last line to get lastWriteAt
           let lastWriteAt = 0;
           if (existsSync(CHECKPOINTS_PATH)) {
             const text = readFileSync(CHECKPOINTS_PATH, 'utf8').trim();
             const lines = text.split('\n');
             const last = lines[lines.length - 1];
             if (last) {
               try {
                 const parsed = JSON.parse(last);
                 lastWriteAt = Date.parse(parsed.ts) || 0;
               } catch { /* malformed — treat as 0 */ }
             }

           if (Date.now() - lastWriteAt < THROTTLE_MS) return;  // throttle

           // Append new line
           mkdirSync(dirname(CHECKPOINTS_PATH), { recursive: true });
           const entry = JSON.stringify({
             ts: new Date().toISOString(),
             tool: input.tool ?? 'unknown',
           });
           appendFileSync(CHECKPOINTS_PATH, entry + '\n', 'utf8');
         } catch {
           // silent — hooks must not crash session
         }
       }
       ```
