**Atomic write — D-07 LOCKED chokepoint (REVIEWS CONVERGENCE — Codex HIGH "atomic-write fsync claim mismatch"):**

`bin/lib/atomic-write.ts` (locked in Plan 03) is THE ONLY module that performs durable atomic writes in pensmith. It internally does: `writeFile(`${path}.tmp`, content) → open(`${path}.tmp`) → fsync(fd) → close(fd) → rename(`${path}.tmp`, path) → fsync(dir)`. The dir-fsync is required on POSIX for the rename to be durable across power loss.

handoff.ts MUST delegate to this — it does NOT reimplement the pattern:

```typescript
import { atomicWrite } from './atomic-write.js';
// ...
await atomicWriteFile(HANDOFF_PATH, content);
```

ESLint rule from Plan 03 (no-restricted-imports / chokepoint enforcement) blocks any other module from importing `node:fs/promises` `writeFile + rename` pair — atomic-write.ts is the single source of fsync truth.

       import { dirname } from 'node:path';
       import { lock } from 'proper-lockfile';
       import { HandoffSchema, type Handoff, HANDOFF_MAX_BYTES } from './schemas/handoff.js';
       import { atomicWrite } from './atomic-write.js';  // D-07 LOCKED chokepoint — see Plan 03

       export const HANDOFF_PATH = '.paper/HANDOFF.json';
       const HANDOFF_LOCK_PATH = '.paper/HANDOFF.json.lock';

       /**
        * CYCLE-2 H-3 REVIEWS CONVERGENCE — D-17 LOCKED shape:
        *   schema_version: 1 (number, NOT '1' string)
        *   last_updated:   ISO datetime (stamped here)
        *   current_section: bare slug or null
        *   phase:          enum
        *   next_action:    human-readable string ≤ 200 chars
        *   breadcrumbs[]:  ≤ 5 (caller-supplied; this function does NOT trim — caller already trimmed)
        *   section_pointers[]: per-section snapshot of slug + paths + state
        */
       export function assembleHandoff(input: {
         phase: Handoff['phase'];
         currentSection: string | null;
         nextAction: string;
         breadcrumbs: Handoff['breadcrumbs'];
         sectionPointers: Handoff['section_pointers'];
       }): Handoff {
         const candidate = {
           schema_version: 1 as const,
           last_updated: new Date().toISOString(),
           current_section: input.currentSection,
           phase: input.phase,
           next_action: input.nextAction.slice(0, 200),  // hard-cap per D-17 LOCKED max(200)
           breadcrumbs: input.breadcrumbs.slice(-5),     // last 5 only (ARCH-04 budget)
           section_pointers: input.sectionPointers,
         };
         return HandoffSchema.parse(candidate);  // throws if total serialized size > 5120 via refine
       }

       export async function writeHandoff(handoff: Handoff): Promise<void> {
         // Schema-validate (defense-in-depth — caller should already have done this)
         HandoffSchema.parse(handoff);
         const content = JSON.stringify(handoff, null, 2);
         if (Buffer.byteLength(content, 'utf8') > HANDOFF_MAX_BYTES) {
           throw new Error(`HANDOFF serialized size ${Buffer.byteLength(content, 'utf8')} exceeds ${HANDOFF_MAX_BYTES} bytes (D-17)`);
         }
         await mkdir(dirname(HANDOFF_PATH), { recursive: true });
         // Lock on a dedicated sentinel file (NOT on HANDOFF.json which may not exist
         // on the first pre-compact run). proper-lockfile.lock(target) requires the
         // target to exist on most platforms; sentinel touch is the portable solution.
         await writeFile(HANDOFF_LOCK_PATH, '', { flag: 'a' });  // touch idempotently
         const release = await lock(HANDOFF_LOCK_PATH, {
           retries: { retries: 5, minTimeout: 50 },
         });
         try {
           // REVIEWS CONVERGENCE (D-07): DELEGATE to atomic-write chokepoint — do NOT
           // reimplement writeFile+rename here. atomic-write.ts handles tmp-write +
           // fsync(fd) + rename + fsync(dir) — the full durable atomic write protocol.
           // Codex HIGH "atomic-write fsync claim mismatch" closed by this delegation.
           await atomicWriteFile(HANDOFF_PATH, content);
         } finally {
           await release();
         }
       }
