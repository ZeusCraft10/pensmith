    ```typescript
    /**
     * CHOKEPOINT — single allowed importer of pdf-parse (D-06, T-3-11).
     * ESLint forbids `pdf-parse` and `pdf-parse/lib/pdf-parse.js` imports
     * outside this file (see eslint.config.js per-file override).
     *
     * Subpath import sidesteps the index.js debug-mode ENOENT bug present
     * in pdf-parse@1.1.1 when consumed from ESM (RESEARCH.md pitfall #1).
     */
    import pdfParse from 'pdf-parse/lib/pdf-parse.js';

    export async function extractPdfText(buf: Buffer | Uint8Array): Promise<string> {
      if (!(buf instanceof Uint8Array) && !Buffer.isBuffer(buf)) {
        throw new TypeError('extractPdfText: input must be Buffer or Uint8Array (no filesystem access from this chokepoint)');
      }
      const input = Buffer.isBuffer(buf) ? buf : Buffer.from(buf);
      try {
        const result = await pdfParse(input);
        return result.text;
      } catch (err: unknown) {
        const msg = err instanceof Error ? err.message : String(err);
        if (msg.includes('ENOENT') && msg.includes('05-versions-space.pdf')) {
          throw new Error('pdf-parse debug-shim ENOENT — confirm import path is `pdf-parse/lib/pdf-parse.js` not bare `pdf-parse`');
        }
        throw err;
      }
    }
    ```

    ```typescript
    /**
     * CHOKEPOINT — single allowed importer of citation-js (D-19).
     * ESLint forbids `citation-js` imports outside this file (eslint.config.js per-file override).
     *
     * BibTeX parsing pitfall: citation-js >= 0.7 lazy-loads CSL templates.
     * Custom apa.csl must be registered via plugins.config.get('@csl').templates.add(...)
     * BEFORE the first cite.format() call (RESEARCH.md pitfall #4).
     *
     * Renders bibliographies only — inline cites are Pandoc's job at compile time
     * (DRAFT.md uses [@citekey] tokens per D-21).
     */
    import { Cite, plugins } from 'citation-js';

    // CYCLE-3 NEW-H-1 REVIEWS CONVERGENCE — re-export Cite so downstream chokepoint
    // consumers (bin/lib/bibtex-write.ts in Plan 04, tests/bibtex-write.test.ts in
    // Plan 09) can `import { Cite } from './citations.js'` WITHOUT ever importing
    // 'citation-js' directly. The D-19 LOCKED chokepoint stays intact: this file
    // remains the SOLE module in the codebase whose source contains `from 'citation-js'`.
    export { Cite };

    let apaRegistered = false;

    function ensureApaTemplate(cslTemplateString: string): void {
      if (apaRegistered) return;
      const config = plugins.config.get('@csl');
      config.templates.add('pensmith-apa', cslTemplateString);
      apaRegistered = true;
    }

    export function parseBibtex(bibtex: string): Array<Record<string, unknown>> {
      try {
        const cite = new Cite(bibtex, { forceType: '@bibtex/text' });
        return cite.data as Array<Record<string, unknown>>;
      } catch (err) {
        const msg = err instanceof Error ? err.message : String(err);
        throw new Error(`parseBibtex: invalid BibTeX — ${msg}`);
      }
    }

    export function renderApa(bibtex: string, cslTemplateString: string): string {
      ensureApaTemplate(cslTemplateString);
      const cite = new Cite(bibtex, { forceType: '@bibtex/text' });
      return cite.format('bibliography', {
        format: 'text',
        template: 'pensmith-apa',
        lang: 'en-US',
      });
    }
    ```

    Note: The implementation registers apa.csl once per process (module-level memoization via `apaRegistered`). The CSL template string is passed by the caller (apa.csl content read from disk by the caller, not by citations.ts — keeps the chokepoint pure-function and FS-free).
