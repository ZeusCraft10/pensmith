Wave 2 — implement bin/lib/atomic-write.ts as the atomic-write chokepoint per ARCH-05 / D-07 / D-04. This is the SECOND foundational library after paths. State (W10), library (W10), checkpoint (W10), session-log (W9), and http cache (W5) ALL persist their JSON via this module — no direct `fs.writeFile` allowed anywhere else (the W0 chokepoint rule enforces it).

Purpose: A naive `fs.writeFile(target, data)` is not crash-safe — if the process dies mid-write, the target is left partially overwritten, corrupting state. The contract per RESEARCH §RQ-8 + D-04 is: `write to {target}.{nonce}.tmp → fsync(tmp) → rename(tmp, target) → fsync(parentDirFd)`. POSIX rename is atomic; NTFS rename is atomic; the fsync(dirFd) flushes the directory entry to disk for durability across power-loss.

The Windows pitfall (RESEARCH §RQ-8 Pitfall A): `fs.openSync(dir, 'r')` then `fs.fsyncSync(dirFd)` raises EPERM on Windows. Per D-04, EPERM on dirFd-fsync is BEST-EFFORT — we swallow it. Rename atomicity is sufficient on NTFS; dir-fsync is the durability cherry-on-top, not a correctness requirement. (Other errors re-throw.)

Output: ~80-120 LoC TypeScript module exporting `atomicWriteFile` (and `atomicAppendFile` for the session log's append-only case), plus two test files.

Algorithm (atomicWriteFile):
1. dir = path.dirname(targetPath); ensure dir exists (mkdir -p).
2. nonce = crypto.randomBytes(6).toString('hex').
3. tmpPath = `${targetPath}.${nonce}.tmp`.
4. Open tmpPath with O_WRONLY|O_CREAT|O_EXCL, mode 0o644.
5. Write data; close fd; if opts.fsync !== false, fsync the fd before close.
6. rename(tmpPath, targetPath).
7. If opts.fsync !== false: open dirfd; fsync(dirfd); close dirfd. Catch EPERM → swallow. Catch ENOSYS → swallow (Linux on tmpfs). Re-throw other errors.
8. On any error before rename: best-effort unlink(tmpPath); re-throw original.

Algorithm (atomicAppendFile):
1. mkdir -p dirname.
2. fd = open(targetPath, O_WRONLY|O_CREAT|O_APPEND, mode).
3. Write line (caller responsible for trailing newline).
4. fsync(fd) if opts.fsync !== false.
5. close fd.

```ts
// bin/lib/atomic-write.ts — atomic-write chokepoint per ARCH-05 / D-07 / D-04.
// Crash-safe write contract:
//   1. write to {target}.{nonce}.tmp via O_WRONLY|O_CREAT|O_EXCL
//   2. fsync(tmp_fd)
//   3. rename(tmp, target)        ← POSIX/NTFS rename is atomic
//   4. fsync(dir_fd)              ← durability across power-loss
//
// Windows quirk (RESEARCH §RQ-8 Pitfall A, D-04):
//   Step 4 raises EPERM on Windows. We swallow EPERM (rename is sufficient
//   for atomicity on NTFS). All other fsync errors re-throw.
//
// Linux tmpfs quirk: ENOSYS from fsync on tmpfs/9p — also swallow.
//
// This is the SOLE file in the repo allowed to call fs.writeFile-family
// functions. The eslint chokepoint at eslint.config.js (D-07) enforces it
// via a per-file exemption forward-declared in W0.

import { randomBytes } from 'node:crypto';
import * as fs from 'node:fs';
import * as fsp from 'node:fs/promises';
import * as path from 'node:path';

export interface AtomicWriteOptions {
  encoding?: BufferEncoding;
  mode?: number;
  fsync?: boolean;
}

const DEFAULT_MODE = 0o644;

async function syncDir(dir: string): Promise<void> {
  let dirFd: fs.promises.FileHandle | undefined;
  try {
    dirFd = await fsp.open(dir, 'r');
    await dirFd.sync();
  } catch (err) {
    const code = (err as NodeJS.ErrnoException)?.code;
    // Windows: directory fsync raises EPERM. NTFS rename is atomic — accept.
    // Some Linux fs (tmpfs / overlayfs / 9p) return ENOSYS — accept.
    if (code === 'EPERM' || code === 'ENOSYS' || code === 'EISDIR') {
      return;
    }
    throw err;
  } finally {
    if (dirFd) await dirFd.close().catch(() => {});
  }
}

export async function atomicWriteFile(
  targetPath: string,
  data: string | Buffer,
  opts: AtomicWriteOptions = {},
): Promise<void> {
  const dir = path.dirname(targetPath);
  await fsp.mkdir(dir, { recursive: true });
  const nonce = randomBytes(6).toString('hex');
  const tmpPath = `${targetPath}.${nonce}.tmp`;
  const mode = opts.mode ?? DEFAULT_MODE;
  const doFsync = opts.fsync !== false;
  const encoding = opts.encoding ?? 'utf8';

  let fh: fs.promises.FileHandle | undefined;
  try {
    fh = await fsp.open(tmpPath, 'wx', mode);
    if (typeof data === 'string') {
      await fh.writeFile(data, { encoding });
    } else {
      await fh.writeFile(data);
    }
    if (doFsync) {
      await fh.sync();
    }
    await fh.close();
    fh = undefined;
    await fsp.rename(tmpPath, targetPath);
    if (doFsync) {
      await syncDir(dir);
    }
  } catch (err) {
    if (fh) {
      await fh.close().catch(() => {});
    }
    // Cross-device rename: copy + unlink fallback.
    if ((err as NodeJS.ErrnoException)?.code === 'EXDEV') {
      try {
        await fsp.copyFile(tmpPath, targetPath);
        await fsp.unlink(tmpPath);
        if (doFsync) await syncDir(dir);
        return;
      } catch (fbErr) {
        await fsp.unlink(tmpPath).catch(() => {});
        throw fbErr;
      }
    }
    await fsp.unlink(tmpPath).catch(() => {});
    throw err;
  }
}

export async function atomicAppendFile(
  targetPath: string,
  line: string,
  opts: AtomicWriteOptions = {},
): Promise<void> {
  const dir = path.dirname(targetPath);
  await fsp.mkdir(dir, { recursive: true });
  const mode = opts.mode ?? DEFAULT_MODE;
  const doFsync = opts.fsync !== false;
  const encoding = opts.encoding ?? 'utf8';
  // O_WRONLY | O_CREAT | O_APPEND — POSIX guarantees atomicity for writes
  // <= PIPE_BUF bytes. Session log's per-line records stay well under 4KB.
  const fh = await fsp.open(targetPath, 'a', mode);
  try {
    await fh.writeFile(line, { encoding });
    if (doFsync) await fh.sync();
  } finally {
    await fh.close();
  }
}
```

```ts
// tests/atomic-write.test.ts
import test from 'node:test';
import assert from 'node:assert/strict';
import * as fsp from 'node:fs/promises';
import * as path from 'node:path';
import * as os from 'node:os';
import { atomicWriteFile, atomicAppendFile } from '../bin/lib/atomic-write.ts';

async function withTmp(fn: (dir: string) => Promise<void>): Promise<void> {
  const dir = await fsp.mkdtemp(path.join(os.tmpdir(), 'pensmith-aw-'));
  try {
    await fn(dir);
  } finally {
    await fsp.rm(dir, { recursive: true, force: true });
  }
}

test('atomicWriteFile round-trips utf8 string', async () => {
  await withTmp(async (dir) => {
    const target = path.join(dir, 'a.json');
    await atomicWriteFile(target, '{"a":1}');
    const got = await fsp.readFile(target, 'utf8');
    assert.equal(got, '{"a":1}');
  });
});

test('atomicWriteFile round-trips Buffer', async () => {
  await withTmp(async (dir) => {
    const target = path.join(dir, 'a.bin');
    const buf = Buffer.from([1, 2, 3, 4]);
    await atomicWriteFile(target, buf);
    const got = await fsp.readFile(target);
    assert.deepEqual(Array.from(got), [1, 2, 3, 4]);
  });
});

test('atomicWriteFile creates missing parent dirs', async () => {
  await withTmp(async (dir) => {
    const target = path.join(dir, 'a', 'b', 'c', 'file.json');
    await atomicWriteFile(target, '{"x":2}');
    const got = await fsp.readFile(target, 'utf8');
    assert.equal(got, '{"x":2}');
  });
});

test('atomicWriteFile leaves no .tmp file on success', async () => {
  await withTmp(async (dir) => {
    const target = path.join(dir, 'a.json');
    await atomicWriteFile(target, '{"a":1}');
    const entries = await fsp.readdir(dir);
    assert.deepEqual(entries.filter(e => e.endsWith('.tmp')), [], `stale tmp: ${entries.join(', ')}`);
  });
});

test('atomicWriteFile preserves old content on rename failure', async () => {
  await withTmp(async (dir) => {
    const target = path.join(dir, 'a.json');
    await atomicWriteFile(target, 'OLD');
    // Force failure by making target dir read-only via a non-existent path target.
    // Simpler: monkey-patch by making the file an unwriteable mode? On Windows
    // chmod is a no-op. Strategy: write to a path whose parent does NOT exist
    // AFTER initial write (rmdir parent before retry — too racy).
    // Robust approach: writing with opts.mode triggers permission check but
    // not failure on most platforms. Skip on Windows; on POSIX, set parent
    // dir to 0o500 (read+exec only), attempt write, expect throw.
    if (process.platform === 'win32') return;
    await fsp.chmod(dir, 0o500);
    let threw = false;
    try {
      await atomicWriteFile(target, 'NEW');
    } catch {
      threw = true;
    }
    await fsp.chmod(dir, 0o755); // restore for cleanup
    if (threw) {
      // OLD must still be readable
      const got = await fsp.readFile(target, 'utf8');
      assert.equal(got, 'OLD');
      const entries = await fsp.readdir(dir);
      assert.deepEqual(entries.filter(e => e.endsWith('.tmp')), [], 'stale tmp leak');
    }
  });
});

test('atomicAppendFile preserves both records (serial)', async () => {
  await withTmp(async (dir) => {
    const log = path.join(dir, 'l.jsonl');
    await atomicAppendFile(log, 'one\n');
    await atomicAppendFile(log, 'two\n');
    const got = await fsp.readFile(log, 'utf8');
    assert.equal(got, 'one\ntwo\n');
  });
});

test('atomicAppendFile creates parent dir', async () => {
  await withTmp(async (dir) => {
    const log = path.join(dir, 'sub', 'l.jsonl');
    await atomicAppendFile(log, 'x\n');
    const got = await fsp.readFile(log, 'utf8');
    assert.equal(got, 'x\n');
  });
});

test('atomicWriteFile with fsync=false skips fsync (smoke)', async () => {
  await withTmp(async (dir) => {
    const target = path.join(dir, 'a.json');
    await atomicWriteFile(target, '{"a":1}', { fsync: false });
    const got = await fsp.readFile(target, 'utf8');
    assert.equal(got, '{"a":1}');
  });
});
```
