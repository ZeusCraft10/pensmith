Lock-file path resolution:
```ts
import { createHash } from 'node:crypto';
import path from 'node:path';
import fsp from 'node:fs/promises';
import { pensmithLockDir } from './paths.ts';

async function lockFilePath(resource: string): Promise<string> {
  const dir = pensmithLockDir();
  await fsp.mkdir(dir, { recursive: true });
  const hash = createHash('sha256').update(resource).digest('hex').slice(0, 12);
  // proper-lockfile creates `{file}.lock`. We give it a target file path of
  // {lockDir}/{hash} (which we touch to existence first) and proper-lockfile
  // makes {hash}.lock alongside it.
  const stub = path.join(dir, hash);
  await fsp.writeFile(stub, '', { flag: 'a' });  // ensure stub exists
  return stub;
}
```

NOTE: proper-lockfile's `lock(file, opts)` requires the file to EXIST. We touch a per-resource stub file then call `lockfile.lock(stubPath, { ... })`.

```ts
// bin/lib/lock.ts — cross-process advisory lock per ARCH-06 / D-26.
// Wraps proper-lockfile@^4 (battle-tested, used by npm/yarn).
//
// CJS-shim necessity (RESEARCH §Key Finding #4 — BLOCKING):
//   `import lockfile from 'proper-lockfile'` raises ERR_REQUIRE_ESM under
//   tsx + node:test in the project's "type":"module" mode. The shim below
//   is the ONLY way to import proper-lockfile under our toolchain.
//
// Lock dir (D-40):
//   Locks live in pensmithLockDir() (~/Library/Application Support/pensmith/locks
//   on macOS; %LOCALAPPDATA%\pensmith\locks on Windows; etc.). NOT inside
//   .paper/ — OneDrive corrupts locks in-tree.

import { createRequire } from 'node:module';
import { createHash } from 'node:crypto';
import path from 'node:path';
import * as fsp from 'node:fs/promises';
import { pensmithLockDir } from './paths.ts';

const require_ = createRequire(import.meta.url);
const lockfile = require_('proper-lockfile') as typeof import('proper-lockfile');

export interface LockOptions {
  timeoutMs?: number;
  staleMs?: number;
  retryDelayMs?: number;
  retryFactor?: number;
}

const DEFAULT_OPTS: Required<LockOptions> = {
  timeoutMs: 60_000,
  staleMs: 45_000,    // 1.5x of the 30s TTL per D-26
  retryDelayMs: 100,
  retryFactor: 1.5,
};

async function stubFor(resource: string): Promise<string> {
  const dir = pensmithLockDir();
  await fsp.mkdir(dir, { recursive: true });
  const hash = createHash('sha256').update(resource).digest('hex').slice(0, 12);
  const stub = path.join(dir, hash);
  // Ensure stub exists; proper-lockfile requires the target to exist.
  // Use fs.promises.open with 'a' flag to create-if-missing without truncating.
  const fh = await fsp.open(stub, 'a');
  await fh.close();
  return stub;
}

function buildPlfOpts(opts: LockOptions = {}) {
  const o = { ...DEFAULT_OPTS, ...opts };
  return {
    stale: o.staleMs,
    retries: {
      retries: Math.max(1, Math.floor(Math.log(o.timeoutMs / o.retryDelayMs) / Math.log(o.retryFactor))),
      minTimeout: o.retryDelayMs,
      maxTimeout: o.timeoutMs,
      factor: o.retryFactor,
    },
  };
}

export async function tryAcquire(
  resource: string,
  opts: LockOptions = {},
): Promise<() => Promise<void>> {
  const stub = await stubFor(resource);
  const release = await lockfile.lock(stub, buildPlfOpts(opts));
  return release;
}

export async function release(resource: string): Promise<void> {
  const stub = await stubFor(resource);
  await lockfile.unlock(stub);
}

export async function isLocked(resource: string): Promise<boolean> {
  const stub = await stubFor(resource);
  return lockfile.check(stub);
}

export async function withLock<T>(
  resource: string,
  fn: () => Promise<T>,
  opts: LockOptions = {},
): Promise<T> {
  const release = await tryAcquire(resource, opts);
  try {
    return await fn();
  } finally {
    await release().catch(() => {
      // The lock is gone or never acquired — nothing to do.
    });
  }
}
```

**Step 2.1 — Create `tests/lock-conflict.cjs`** as a CommonJS helper. Because the test runs under tsx, but child_process.spawn invokes Node directly, we use `.cjs` so it can `require('proper-lockfile')` natively.

Skeleton:

```js
// tests/lock-conflict.cjs
// Child helper for the cross-process lock conflict test (TEST-07).
// Spawned by tests/lock.test.ts via child_process.spawn; reads RESOURCE
// and HOLD_MS from env; acquires the same lock the parent will race for;
// holds for HOLD_MS; releases; exits 0.
//
// We deliberately use raw proper-lockfile here (not bin/lib/lock.ts) to
// keep this helper toolchain-independent and avoid an ESM/tsx transitive
// import. The lock-file path computation MUST match bin/lib/lock.ts —
// sha256(resource).slice(0,12) under pensmithLockDir().

const path = require('node:path');
const fs = require('node:fs');
const os = require('node:os');
const { createHash } = require('node:crypto');
const lockfile = require('proper-lockfile');

function localDataDir() {
  if (process.platform === 'win32') {
    if (!process.env.LOCALAPPDATA) throw new Error('LOCALAPPDATA unset');
    return process.env.LOCALAPPDATA;
  }
  if (process.platform === 'darwin') {
    return path.join(process.env.HOME || os.homedir(), 'Library', 'Application Support');
  }
  return process.env.XDG_DATA_HOME || path.join(process.env.HOME || os.homedir(), '.local', 'share');
}

function pensmithLockDir() {
  return path.join(localDataDir(), 'pensmith', 'locks');
}

function stubFor(resource) {
  const dir = pensmithLockDir();
  fs.mkdirSync(dir, { recursive: true });
  const hash = createHash('sha256').update(resource).digest('hex').slice(0, 12);
  const stub = path.join(dir, hash);
  fs.closeSync(fs.openSync(stub, 'a'));
  return stub;
}

(async () => {
  const resource = process.env.RESOURCE;
  const holdMs = parseInt(process.env.HOLD_MS || '2000', 10);
  if (!resource) {
    console.error('RESOURCE env required');
    process.exit(2);
  }
  const stub = stubFor(resource);
  const release = await lockfile.lock(stub, { stale: 45_000, retries: 0 });
  console.log('ACQUIRED', Date.now());
  await new Promise(r => setTimeout(r, holdMs));
  console.log('RELEASING', Date.now());
  await release();
  process.exit(0);
})().catch((err) => {
  console.error('CHILD-ERROR', err.message);
  process.exit(3);
});
```

```ts
// tests/lock.test.ts
import test from 'node:test';
import assert from 'node:assert/strict';
import { spawn } from 'node:child_process';
import path from 'node:path';
import { withLock, tryAcquire, isLocked } from '../bin/lib/lock.ts';

test('withLock returns inner value', async () => {
  const v = await withLock('test:rv:' + Date.now(), async () => 42);
  assert.equal(v, 42);
});

test('withLock releases on inner throw', async () => {
  const r = 'test:throw:' + Date.now();
  await assert.rejects(withLock(r, async () => { throw new Error('boom'); }), /boom/);
  // After: isLocked must be false
  assert.equal(await isLocked(r), false);
});

test('tryAcquire / release / isLocked roundtrip', async () => {
  const r = 'test:try:' + Date.now();
  assert.equal(await isLocked(r), false);
  const release = await tryAcquire(r);
  assert.equal(await isLocked(r), true);
  await release();
  assert.equal(await isLocked(r), false);
});

test('serialized within same process — second withLock waits for first', async () => {
  const r = 'test:serial:' + Date.now();
  const order: string[] = [];
  const a = withLock(r, async () => {
    order.push('A-start');
    await new Promise(res => setTimeout(res, 100));
    order.push('A-end');
    return 'a';
  });
  // small delay to ensure A starts first
  await new Promise(res => setTimeout(res, 10));
  const b = withLock(r, async () => {
    order.push('B-start');
    return 'b';
  });
  const [va, vb] = await Promise.all([a, b]);
  assert.equal(va, 'a');
  assert.equal(vb, 'b');
  assert.deepEqual(order, ['A-start', 'A-end', 'B-start']);
});

test('cross-process: child holds, parent waits (TEST-07)', async () => {
  const r = 'test:xp:' + Date.now();
  const helper = path.resolve('tests/lock-conflict.cjs');
  const HOLD_MS = 1500;
  const child = spawn(process.execPath, [helper], {
    env: { ...process.env, RESOURCE: r, HOLD_MS: String(HOLD_MS) },
    stdio: ['ignore', 'pipe', 'pipe'],
  });
  let childAcquiredAt = 0;
  child.stdout.on('data', (chunk) => {
    const line = chunk.toString();
    const m = line.match(/ACQUIRED (\d+)/);
    if (m) childAcquiredAt = parseInt(m[1], 10);
  });
  // Wait for child to acquire
  await new Promise<void>((res, rej) => {
    const t = setTimeout(() => rej(new Error('child failed to acquire within 3s')), 3000);
    const i = setInterval(() => {
      if (childAcquiredAt) { clearInterval(i); clearTimeout(t); res(); }
    }, 50);
  });
  // Parent attempts to acquire — must wait for child to release
  const parentStart = Date.now();
  const v = await withLock(r, async () => 'parent', { timeoutMs: 10_000, staleMs: 45_000 });
  const parentAcquiredAt = Date.now();
  assert.equal(v, 'parent');
  // Parent should have waited at least HOLD_MS - small slack since child started a bit before parent.
  const waitMs = parentAcquiredAt - parentStart;
  assert.ok(waitMs >= HOLD_MS - 500,
    `parent should have waited ~${HOLD_MS}ms; actually waited ${waitMs}ms`);
  // Drain child
  await new Promise<void>((res) => child.on('exit', () => res()));
});
```

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |

| T-01-02 | Tampering | a stale lock from a crashed process blocks new acquisitions forever | mitigate | proper-lockfile's `stale: 45_000` (1.5x of the 30s TTL per D-26) auto-recovers locks older than 45s. Crashed-process recovery test is implicit in stale handling. |

| T-01-DOS-02 | DoS | attacker creates 10000 fake stub files in pensmithLockDir | accept | per-user data dir, only the user's own processes can write there; DoS would be self-inflicted |
