export type Kind =
  | 'prompt'
  | 'response'
  | 'tool_call'
  | 'tool_result'
  | 'cost'
  | 'event'
  | 'warn'
  | 'error';

// Each method takes a free-form payload object. The writer redacts strings/objects
// before serialization. The kind discriminator is set by which method was called.
export interface SessionLogger {
  prompt(payload: Record<string, unknown>): void;
  response(payload: Record<string, unknown>): void;
  toolCall(payload: Record<string, unknown>): void;     // wire `kind` = 'tool_call'
  toolResult(payload: Record<string, unknown>): void;   // wire `kind` = 'tool_result'
  cost(payload: Record<string, unknown>): void;
  event(payload: Record<string, unknown>): void;
  warn(payload: Record<string, unknown>): void;
  error(payload: Record<string, unknown>): void;
  child(bindings: Record<string, unknown>): SessionLogger;
  close(): Promise<void>;
}

export interface OpenSessionLogOptions {
  scope?: 'paper' | 'global' | 'auto';   // default 'auto' — paper if cwd is in a paper, else global
  cwd?: string;                          // override for tests
  maxBytes?: number;                     // default 50 * 1024 * 1024 (D-51)
  maxBackups?: number;                   // default 3 (D-51 — last 3 rotated files kept)
  maxRecordBytes?: number;               // default 16 * 1024 (D-50)
}

export function openSessionLog(opts?: OpenSessionLogOptions): SessionLogger;

// D-52 toggle hook — module-scope setter wired by Phase 7's CLI.
export function setMirrorPromptsToStderr(enabled: boolean): void;

    **Record construction (D-49 — `{at, kind, run_id, ...payload}`):**

    ```typescript
    interface BaseRecord {
      at: string;          // ISO-8601 — new Date().toISOString()
      kind: Kind;
      run_id: string;
      // payload spread inline — no `ctx`/`msg` wrapper
      [key: string]: unknown;
    }

    function buildRecord(
      kind: Kind,
      payload: Record<string, unknown>,
      bindings: Record<string, unknown>,
      run_id: string,
    ): BaseRecord {
      // Merge: bindings → payload (payload wins). Both go through redaction before merge:
      //   - object-valued fields: redactKeys (recurses into the whole object)
      //   - string-valued fields: redactPii on each string
      const merged = { ...bindings, ...payload };
      const safe = redactKeys(merged) as Record<string, unknown>;
      // Walk safe; redactPii any string leaves
      for (const k of Object.keys(safe)) {
        const v = safe[k];
        if (typeof v === 'string') safe[k] = redactPii(v);
      }
      return {
        at: new Date().toISOString(),
        kind,
        run_id,
        ...safe,
      };
    }
    ```

    **Truncation + spillover (D-50):**

    ```typescript
    async function writeLineOrTruncate(
      logFile: string,
      spillRoot: string,
      run_id: string,
      seqRef: { value: number },
      record: BaseRecord,
      maxRecordBytes: number,
    ): Promise<string> {
      const line = JSON.stringify(record) + '\n';
      const sizeBytes = Buffer.byteLength(line, 'utf8');
      if (sizeBytes <= maxRecordBytes) return line;

      // Oversize: separate `at`, `kind`, `run_id` from payload
      const { at, kind, run_id: rid, ...payload } = record;
      const seq = seqRef.value++;
      const spillFile = path.join(spillRoot, run_id, `${seq}.json`);
      try {
        await fs.promises.mkdir(path.dirname(spillFile), { recursive: true });
        await atomicWriteFile(spillFile, JSON.stringify(record, null, 2) + '\n');
      } catch { /* spill is best-effort; truncated line still written */ }

      const stringified = JSON.stringify(payload);
      const head = stringified.slice(0, HEAD_TAIL_BYTES);
      const tail = stringified.slice(-HEAD_TAIL_BYTES);
      // Path written into the line is RELATIVE to the log file's parent so it's grep-friendly
      const spillRel = `sessions/${run_id}/${seq}.json`;
      const truncated: BaseRecord = {
        at,
        kind,
        run_id: rid,
        head,
        tail,
        truncated: true,
        spilled_to: spillRel,
      };
      return JSON.stringify(truncated) + '\n';
    }
    ```

    **Rotation (D-51 — 50MB threshold, 3 backups):**

    ```typescript
    async function maybeRotate(filePath: string, maxBytes: number, maxBackups: number) {
      try {
        const st = await fs.promises.stat(filePath);
        if (st.size <= maxBytes) return;
        // Shift backups: unlink oldest, then rename .{n-1} → .{n}, finally current → .1
        for (let i = maxBackups; i >= 1; i--) {
          const src = i === 1 ? filePath : `${filePath}.${i - 1}`;
          const dst = `${filePath}.${i}`;
          if (i === maxBackups) {
            try { await fs.promises.unlink(dst); } catch { /* ENOENT ok */ }
          }
          try { await fs.promises.rename(src, dst); } catch { /* ENOENT ok if a backup slot empty */ }
        }
      } catch { /* swallow — logging must never throw */ }
    }
    ```

    With maxBackups=3: `unlink .3`, `rename .2 → .3`, `rename .1 → .2`, `rename current → .1`. Order matters (highest-numbered first) and unlinking the oldest target before final rename is Windows-safe.

    ```typescript
    export function openSessionLog(opts: OpenSessionLogOptions = {}): SessionLogger {
      const scope = opts.scope ?? 'auto';
      const cwd = opts.cwd ?? process.cwd();
      const maxBytes = opts.maxBytes ?? MAX_LOG_BYTES;
      const maxBackups = opts.maxBackups ?? MAX_BACKUPS;
      const maxRecordBytes = opts.maxRecordBytes ?? MAX_RECORD_BYTES;
      const { logFile, spillRoot } = resolveRoot({ scope, cwd });
      // run_id: per-handle unique identifier. Per D-64 we have no `ulid` dep — use
      // crypto.randomUUID() (UUIDv4). RESEARCH §V3 explicitly accepts this.
      const run_id = randomUUID();
      const seqRef = { value: 0 };

      function makeLogger(bindings: Record<string, unknown>): SessionLogger {
        function emit(kind: Kind, payload: Record<string, unknown>) {
          const record = buildRecord(kind, payload, bindings, run_id);
          // Mirror BEFORE async write so it's synchronous from caller's POV
          mirrorIfPrompt(record);
          enqueue(async () => {
            try {
              const line = await writeLineOrTruncate(logFile, spillRoot, run_id, seqRef, record, maxRecordBytes);
              await atomicAppendFile(logFile, line);
              await maybeRotate(logFile, maxBytes, maxBackups);
            } catch { /* swallow */ }
          });
        }
        return {
          prompt:     (p) => emit('prompt',      p),
          response:   (p) => emit('response',    p),
          toolCall:   (p) => emit('tool_call',   p),
          toolResult: (p) => emit('tool_result', p),
          cost:       (p) => emit('cost',        p),
          event:      (p) => emit('event',       p),
          warn:       (p) => emit('warn',        p),
          error:      (p) => emit('error',       p),
          child:      (b) => makeLogger({ ...bindings, ...(redactKeys(b) as Record<string, unknown>) }),
          close: async () => { await chain; },
        };
      }
      return makeLogger({});
    }
    ```
