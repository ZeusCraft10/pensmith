    ```typescript
    export async function initState(paperRoot: string, seed?: { paperId?: string }): Promise<State> {
      const file = stateFile(paperRoot);
      // refuse to clobber
      try {
        await fs.promises.access(file);
        throw new StateAlreadyExistsError(`STATE.json already exists at ${file}`);
      } catch (e: any) {
        if (e?.code !== 'ENOENT' && !(e instanceof StateNotFoundError)) {
          if (e instanceof StateAlreadyExistsError) throw e;
          // Permission errors etc bubble up
          if (e?.code && e.code !== 'ENOENT') throw e;
        }
      }
      const seeded: State = StateSchema.parse({
        $schemaVersion: CURRENT_STATE_VERSION,
        paperId: seed?.paperId ?? randomUUID(),
        createdAt: new Date().toISOString(),
      });
      await withLock(file, async () => {
        await atomicWriteFile(file, JSON.stringify(seeded, null, 2) + '\n');
      });
      log().event({ event: 'state.init', paperId: seeded.paperId, schemaVersion: seeded.$schemaVersion });
      return seeded;
    }
    ```

    **loadState:**

    ```typescript
    export async function loadState(paperRoot: string): Promise<State> {
      const file = stateFile(paperRoot);
      try {
        const value = await loadAndMigrate({
          file,
          schema: StateSchema,
          schemaName: 'state',
          currentVersion: CURRENT_STATE_VERSION,
          writeBack: true,
        });
        log().event({ event: 'state.load', paperId: (value as State).paperId, schemaVersion: (value as State).$schemaVersion });
        return value as State;
      } catch (e: any) {
        if (e?.code === 'ENOENT' || e?.cause?.code === 'ENOENT') {
          throw new StateNotFoundError(`STATE.json not found at ${file}`);
        }
        throw e;
      }
    }
    ```

    **saveState:**

    ```typescript
    export async function saveState(paperRoot: string, state: State): Promise<void> {
      const file = stateFile(paperRoot);
      const validated = StateSchema.parse(state);  // refuse to write garbage
      await withLock(file, async () => {
        await atomicWriteFile(file, JSON.stringify(validated, null, 2) + '\n');
      });
      log().event({ event: 'state.save', paperId: validated.paperId, schemaVersion: validated.$schemaVersion });
    }
    ```

    **updateState — critical that load + mutate + save share ONE lock:**

    ```typescript
    export async function updateState(paperRoot: string, mutator: (s: State) => State | Promise<State>): Promise<State> {
      const file = stateFile(paperRoot);
      let next!: State;
      await withLock(file, async () => {
        const current = await loadAndMigrate({
          file,
          schema: StateSchema,
          schemaName: 'state',
          currentVersion: CURRENT_STATE_VERSION,
          writeBack: false,  // we're going to write a fresh copy momentarily
        }) as State;
        const candidate = await mutator(current);
        next = StateSchema.parse(candidate);
        await atomicWriteFile(file, JSON.stringify(next, null, 2) + '\n');
      });
      log().event({ event: 'state.update', paperId: next.paperId, schemaVersion: next.$schemaVersion });
      return next;
    }
    ```

We test contention by using a transient extension: pass a synthetic `__counter` field through StateSchema.passthrough() — BUT that requires schema cooperation. Alternative (cleaner): test contention by having both updateState callers run a mutator that re-reads, then call saveState with a NEW createdAt; the LATER-locked write wins, and we assert the chain serialized (no torn write).

    Simplest reliable assertion: spawn two updateState calls that both refresh `createdAt` to a known per-task ISO. After both await, the file's `createdAt` equals exactly one of the two; both mutators ran without exception; the file is parseable.

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| concurrent callers within one process or across processes → STATE.json | two writers can race; lock + atomic-write together guarantee no torn state |
| disk-resident STATE.json from older pensmith version → newer code | forward-incompat must REFUSE to clobber-downgrade |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-01-01 | T (Tampering) | concurrent state writes corrupting STATE.json | mitigate | every save/update wrapped in withLock (W3); writes via atomicWriteFile (W2); load happens inside the lock for updateState |
| T-01-COMPAT-01 | I (Information disclosure / silent data loss) | newer-on-disk version silently downgraded | mitigate | loadAndMigrate refuses-forward-incompat (W7); state.ts surfaces ForwardIncompatError unchanged |
| T-01-08 | T (Tampering) | malformed state on disk passes loader | mitigate | StateSchema.parse runs on every load AND save AND update path (defense-in-depth — schema-validated 3x) |
</threat_model>
