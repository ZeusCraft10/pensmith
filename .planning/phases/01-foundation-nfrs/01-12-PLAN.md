    ```typescript
    export async function recordCheckpoint(
      paperRoot: string,
      label: string,
      refs?: Record<string, unknown>,
    ): Promise<Checkpoint> {
      const file = checkpointFile(paperRoot);
      const checkpoint: Checkpoint = CheckpointSchema.parse({
        $schemaVersion: CURRENT_CHECKPOINT_VERSION,
        label,
        tookAt: new Date().toISOString(),
        ...(refs !== undefined ? { refs } : {}),
      });
      const line = JSON.stringify(checkpoint) + '\n';
      await withLock(file, async () => {
        await atomicAppendFile(file, line);
      });
      log().event({ event: 'checkpoint.record', label, schemaVersion: checkpoint.$schemaVersion });
      return checkpoint;
    }
    ```

    Note: per D-04, atomicAppendFile uses O_APPEND so each `write` syscall is atomic for ≤ PIPE_BUF bytes (4 KB on Linux). For lines > 4 KB the lock provides the cross-process guarantee.

    **listCheckpoints — tolerant reader:**

    ```typescript
    export async function listCheckpoints(paperRoot: string): Promise<Checkpoint[]> {
      const file = checkpointFile(paperRoot);
      let raw: string;
      try {
        raw = await fs.promises.readFile(file, 'utf8');
      } catch (e: any) {
        if (e?.code === 'ENOENT') return [];  // empty history
        throw e;
      }
      const lines = raw.split('\n').filter(l => l.length > 0);
      const out: Checkpoint[] = [];
      let skipped = 0;
      for (const ln of lines) {
        let parsed: unknown;
        try { parsed = JSON.parse(ln); }
        catch { skipped++; continue; }
        const result = CheckpointSchema.safeParse(parsed);
        if (result.success) {
          out.push(result.data);
        } else {
          skipped++;
        }
      }
      if (skipped > 0) {
        log().warn({ event: 'checkpoint.list.skipped', skipped, kept: out.length });
      } else {
        log().event({ event: 'checkpoint.list', count: out.length });
      }
      return out;
    }
    ```

    **findCheckpoint — most recent matching label:**

    ```typescript
    export async function findCheckpoint(paperRoot: string, label: string): Promise<Checkpoint | undefined> {
      const all = await listCheckpoints(paperRoot);
      // reverse iteration so we return most recent
      for (let i = all.length - 1; i >= 0; i--) {
        if (all[i].label === label) return all[i];
      }
      return undefined;
    }
    ```

    **Why the forward-compat exception (per D-60 carve-out):**
    State and library are AUTHORITATIVE — newer-on-disk content from a future pensmith version, opened by older code, must NOT be silently downgraded (you'd lose data). Checkpoints are AUDIT history — newer-versioned entries are interesting to a newer reader and uninteresting to an older one. Skipping them in listCheckpoints is the safe behavior because the file is append-only: skipping never causes data loss, and the older reader simply sees an older "history view." Document this explicitly in the SUMMARY.

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| concurrent recordCheckpoint callers | many phases may emit checkpoints simultaneously during research-fanout / verifier-fanout |
| disk-resident CHECKPOINTS.jsonl from older or newer pensmith version → reader code | per D-60 carve-out: newer-versioned lines are SKIPPED rather than rejected |
| arbitrary `refs` payload from caller | foundation slice accepts any record; downstream waves will narrow |

| Threat ID | Category | Component | Disposition | Mitigation Plan |
