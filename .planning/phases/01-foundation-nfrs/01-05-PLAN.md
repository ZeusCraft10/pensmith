**Step 1.1 — Create `bin/lib/retry.ts`** as a thin wrapper around p-retry. Per RESEARCH §RQ-9, p-retry@6 supports `randomize: true` but it's bounded between `randomizeStart` and `randomizeEnd` (default 1.0 to 2.0) — that's bounded multiplicative jitter. AWS full-jitter is `delay = random(0, baseDelay * 2^attempt)`. We compute the per-attempt delay ourselves and tell p-retry to use a custom delay function via `forever: false, retries: maxAttempts-1, factor: 2, minTimeout, maxTimeout, randomize: false` and override with our `onFailedAttempt` hook + manual sleep.

Cleaner approach: hand-roll the loop in the shim, no p-retry dependency. p-retry is still installed (W0 dep) for consumers who want it directly, but retry.ts is the chokepoint.

```ts
// bin/lib/retry.ts — full-jitter retry shim per RESEARCH §RQ-9 + Key Finding #2.
// AWS full-jitter formula:
//   delay(attempt) = random_between(0, min(capMs, baseMs * 2^attempt))
// This decorrelates retries across many clients, preventing thundering-herd
// stampedes that bounded multiplicative jitter (p-retry default) cannot.

export interface RetryOptions {
  maxAttempts?: number;
  baseMs?: number;
  capMs?: number;
  retryOn?: (err: unknown, attempt: number) => boolean;
  onAttempt?: (attempt: number, err: unknown) => void;
}

const DEFAULT_OPTS: Required<Omit<RetryOptions, 'retryOn' | 'onAttempt'>> & {
  retryOn: NonNullable<RetryOptions['retryOn']>;
  onAttempt: NonNullable<RetryOptions['onAttempt']>;
} = {
  maxAttempts: 5,
  baseMs: 200,
  capMs: 30_000,
  retryOn: () => true,
  onAttempt: () => {},
};

export function fullJitterDelayMs(
  attempt: number,        // 1-based
  baseMs: number,
  capMs: number,
): number {
  // Math.min cap, then uniform random [0, cap]
  const exp = Math.min(capMs, baseMs * Math.pow(2, attempt - 1));
  return Math.floor(Math.random() * (exp + 1));
}

const sleep = (ms: number) => new Promise<void>((r) => setTimeout(r, ms));

export async function retry<T>(
  fn: () => Promise<T>,
  opts: RetryOptions = {},
): Promise<T> {
  const o = { ...DEFAULT_OPTS, ...opts };
  let lastErr: unknown;
  for (let attempt = 1; attempt <= o.maxAttempts; attempt++) {
    try {
      return await fn();
    } catch (err) {
      lastErr = err;
      o.onAttempt(attempt, err);
      if (attempt >= o.maxAttempts) break;
      if (!o.retryOn(err, attempt)) break;
      const delay = fullJitterDelayMs(attempt, o.baseMs, o.capMs);
      await sleep(delay);
    }
  }
  throw lastErr;
}
```

```ts
// bin/lib/http.ts — HTTP client chokepoint per ARCH-12 / ARCH-13.
// SOLE call site for undici / node:http / node:https in the repo (D-06).
//
// Polite scholarly client (D-23, D-24):
//   User-Agent: pensmith/{version} ({PENSMITH_CONTACT_EMAIL || 'no-contact'})
//   On missing PENSMITH_CONTACT_EMAIL, WARN-once stderr banner from
//   references/http-warnings.md (locked string).
//
// Per-source TTL disk cache (D-30):
//   crossref/openalex/arxiv/pubmed: 7d, unpaywall: 1d, generic: 24h.
//   Cache key = sha256(method + ':' + url + ':' + sortedHeaders).slice(0,16).
//   Cache write goes through atomicWriteFile (W2 dep) — never direct fs.writeFile.
//
// Retry (D-31, D-32):
//   429/500/502/503/504 + ETIMEDOUT/ECONNRESET/ENOTFOUND → full-jitter retry.

import { request } from 'undici';
import { createHash } from 'node:crypto';
import { readFileSync } from 'node:fs';
import * as fsp from 'node:fs/promises';
import path from 'node:path';
import { fileURLToPath } from 'node:url';
import { pensmithHttpCacheDir } from './paths.ts';
import { atomicWriteFile } from './atomic-write.ts';
import { retry } from './retry.ts';

const __filename = fileURLToPath(import.meta.url);
const __dirname = path.dirname(__filename);

// ---------- WARN-once for missing contact email ----------
const WARN_FILE = path.resolve(__dirname, '..', '..', 'references', 'http-warnings.md');
let warnString: string | null = null;
let warnedNoEmail = false;

function loadWarnString(): string {
  if (warnString !== null) return warnString;
  const md = readFileSync(WARN_FILE, 'utf8');
  // Find the first blockquote line under "## PENSMITH_CONTACT_EMAIL not set"
  const lines = md.split(/\r?\n/);
  let inSection = false;
  for (const line of lines) {
    if (line.startsWith('## PENSMITH_CONTACT_EMAIL not set')) {
      inSection = true;
      continue;
    }
    if (inSection && line.startsWith('> ')) {
      warnString = line.slice(2).trim();
      return warnString;
    }
  }
  warnString = 'pensmith: PENSMITH_CONTACT_EMAIL is not set.';
  return warnString;
}

function warnNoEmailOnce(): void {
  if (warnedNoEmail) return;
  warnedNoEmail = true;
  process.stderr.write(loadWarnString() + '\n');
}

// ---------- User-Agent ----------
let cachedVersion: string | null = null;
function pkgVersion(): string {
  if (cachedVersion !== null) return cachedVersion;
  try {
    const pkgPath = path.resolve(__dirname, '..', '..', 'package.json');
    const pkg = JSON.parse(readFileSync(pkgPath, 'utf8'));
    cachedVersion = (pkg.version as string) ?? '0.0.0';
  } catch {
    cachedVersion = '0.0.0';
  }
  return cachedVersion;
}

function userAgent(): string {
  const email = process.env.PENSMITH_CONTACT_EMAIL?.trim();
  if (!email) {
    warnNoEmailOnce();
    return `pensmith/${pkgVersion()} (no-contact)`;
  }
  return `pensmith/${pkgVersion()} (${email})`;
}

// ---------- TTL table ----------
const TTL_MS_BY_SOURCE: Record<NonNullable<FetchOptions['source']>, number> = {
  crossref: 7 * 24 * 3600_000,
  openalex: 7 * 24 * 3600_000,
  arxiv: 7 * 24 * 3600_000,
  pubmed: 7 * 24 * 3600_000,
  unpaywall: 1 * 24 * 3600_000,
  generic: 24 * 3600_000,
};

// ---------- Cache key ----------
function cacheKey(method: string, url: string, headers: Record<string, string>): string {
  const sorted = Object.entries(headers).sort(([a], [b]) => a.localeCompare(b));
  const headerStr = sorted.map(([k, v]) => `${k.toLowerCase()}:${v}`).join('|');
  return createHash('sha256').update(`${method}:${url}:${headerStr}`).digest('hex').slice(0, 16);
}

async function readCache(key: string, ttlMs: number): Promise<HttpResponse | null> {
  const file = path.join(pensmithHttpCacheDir(), `${key}.json`);
  try {
    const raw = await fsp.readFile(file, 'utf8');
    const parsed = JSON.parse(raw) as { savedAt: string; response: HttpResponse };
    const ageMs = Date.now() - new Date(parsed.savedAt).getTime();
    if (ageMs > ttlMs) return null;
    return { ...parsed.response, cached: true, cachedAt: parsed.savedAt };
  } catch {
    return null;
  }
}

async function writeCache(key: string, response: HttpResponse): Promise<void> {
  const file = path.join(pensmithHttpCacheDir(), `${key}.json`);
  const payload = JSON.stringify({ savedAt: new Date().toISOString(), response });
  await atomicWriteFile(file, payload);
}

export interface HttpResponse {
  status: number;
  headers: Record<string, string>;
  body: string;
  cached: boolean;
  cachedAt?: string;
}

export interface FetchOptions {
  method?: 'GET' | 'POST' | 'HEAD';
  headers?: Record<string, string>;
  body?: string | Buffer;
  source?: 'crossref' | 'openalex' | 'unpaywall' | 'arxiv' | 'pubmed' | 'generic';
  cacheKey?: string;
  cacheTtlMs?: number;
  timeoutMs?: number;
  noCache?: boolean;
  noRetry?: boolean;
}

const DEFAULT_TIMEOUT_MS = 30_000;
const RETRYABLE_STATUSES = new Set([429, 500, 502, 503, 504]);
const RETRYABLE_ERR_CODES = new Set(['ETIMEDOUT', 'ECONNRESET', 'ENOTFOUND', 'EAI_AGAIN']);

export async function fetch(url: string, opts: FetchOptions = {}): Promise<HttpResponse> {
  const method = opts.method ?? 'GET';
  const headers: Record<string, string> = {
    'user-agent': userAgent(),
    accept: 'application/json',
    ...opts.headers,
  };
  const source = opts.source ?? 'generic';
  const ttlMs = opts.cacheTtlMs ?? TTL_MS_BY_SOURCE[source];
  const key = opts.cacheKey ?? cacheKey(method, url, headers);

  // Cache short-circuit (only for GET, only if cache enabled)
  if (method === 'GET' && !opts.noCache) {
    const cached = await readCache(key, ttlMs);
    if (cached) return cached;
  }

  const callOnce = async (): Promise<HttpResponse> => {
    const { statusCode, headers: rh, body } = await request(url, {
      method,
      headers,
      headersTimeout: opts.timeoutMs ?? DEFAULT_TIMEOUT_MS,
      bodyTimeout: opts.timeoutMs ?? DEFAULT_TIMEOUT_MS,
      ...(opts.body !== undefined ? { body: opts.body } : {}),
    });
    const text = await body.text();
    const flatHeaders: Record<string, string> = {};
    for (const [k, v] of Object.entries(rh)) {
      flatHeaders[k.toLowerCase()] = Array.isArray(v) ? v.join(', ') : (v ?? '');
    }
    return { status: statusCode, headers: flatHeaders, body: text, cached: false };
  };

  const response = opts.noRetry
    ? await callOnce()
    : await retry(async () => {
        const r = await callOnce();
        if (RETRYABLE_STATUSES.has(r.status)) {
          // Throw to trigger retry; carry response for diagnostics
          const err = new Error(`HTTP ${r.status}`) as Error & { status?: number };
          err.status = r.status;
          throw err;
        }
        return r;
      }, {
        maxAttempts: 5,
        baseMs: 200,
        capMs: 30_000,
        retryOn: (err) => {
          const e = err as { status?: number; code?: string };
          if (typeof e?.status === 'number' && RETRYABLE_STATUSES.has(e.status)) return true;
          if (typeof e?.code === 'string' && RETRYABLE_ERR_CODES.has(e.code)) return true;
          return false;
        },
      });

  // Cache successful + 404 GETs
  if (method === 'GET' && !opts.noCache && (response.status === 200 || response.status === 404)) {
    await writeCache(key, response).catch(() => {});
  }
  return response;
}

export async function clearCache(): Promise<void> {
  const dir = pensmithHttpCacheDir();
  try {
    const entries = await fsp.readdir(dir);
    await Promise.all(entries.map((f) => fsp.unlink(path.join(dir, f)).catch(() => {})));
  } catch {
    // dir doesn't exist — fine
  }
```

**Step 2.1.5 — Add a per-source TokenBucket pre-dispatch gate (ARCH-13).** Before the cache short-circuit, fetch() acquires a token from a per-source TokenBucket so pensmith never exceeds the polite-pool rates baked into Crossref / OpenAlex / Unpaywall (RESEARCH §RQ-1). Implementation:

```ts
// ---------- Per-source TokenBucket (ARCH-13) ----------
// Polite-pool rates per RESEARCH §RQ-1 (rps = requests per second):
const RPS_BY_SOURCE: Record<NonNullable<FetchOptions['source']>, number> = {
  crossref: 50,    // polite pool — RESEARCH §RQ-1
  openalex: 10,    // anonymous; 100 with key — defensive choice for v0.1
  unpaywall: 10,   // explicit per-key budget — same defensive default
  arxiv:    1,     // 1 req/3s recommended; we relax slightly
  pubmed:   3,     // E-utilities anonymous: 3 rps; 10 with key
  generic:  5,     // untyped fallback
};

class TokenBucket {
  private tokens: number;
  private lastRefillMs: number;
  constructor(private capacity: number, private refillPerSec: number) {
    this.tokens = capacity;
    this.lastRefillMs = Date.now();
  }
  private refill(): void {
    const now = Date.now();
    const elapsedSec = (now - this.lastRefillMs) / 1000;
    this.tokens = Math.min(this.capacity, this.tokens + elapsedSec * this.refillPerSec);
    this.lastRefillMs = now;
  }
  async acquire(): Promise<void> {
    // Loop because setTimeout granularity may leave us slightly short on the
    // first wakeup; we refill again and try.
    // eslint-disable-next-line no-constant-condition
    while (true) {
      this.refill();
      if (this.tokens >= 1) {
        this.tokens -= 1;
        return;
      }
      const deficit = 1 - this.tokens;
      const waitMs = Math.ceil((deficit / this.refillPerSec) * 1000);
      await new Promise((r) => setTimeout(r, Math.max(1, waitMs)));
    }
  }
}

const BUCKETS: Partial<Record<NonNullable<FetchOptions['source']>, TokenBucket>> = {};
function bucketFor(src: NonNullable<FetchOptions['source']>): TokenBucket {
  if (!BUCKETS[src]) {
    const rps = RPS_BY_SOURCE[src];
    // Capacity = rps so that a fresh process has a small burst budget.
    BUCKETS[src] = new TokenBucket(rps, rps);
  }
  return BUCKETS[src]!;
}
```

Wire it into `fetch()` BEFORE the cache short-circuit changes (see existing Step 2.1) — the bucket acquire happens regardless of cache hit so a hot-cache run still respects polite-pool rates if it falls through to network. **However**, to make cache hits truly free, MOVE the `bucketFor(source).acquire()` call to JUST BEFORE `callOnce()` execution (i.e. AFTER the cache short-circuit returns) so cache hits skip the bucket entirely. Final dispatch order:

1. Compute key + headers
2. Cache short-circuit (if hit, return immediately — no bucket needed)
3. `await bucketFor(source).acquire()` — pre-dispatch rate gate
4. `callOnce()` (or retry-wrapped variant)

Acquire BEFORE the retry wrapper, NOT inside it: a 429 retry should re-acquire to be polite, so the retry's `fn` must include the bucket acquire. The simplest shape:

```ts
const dispatch = async (): Promise<HttpResponse> => {
  await bucketFor(source).acquire();
  return callOnce();
};

const response = opts.noRetry
  ? await dispatch()
  : await retry(async () => {
      const r = await dispatch();
      if (RETRYABLE_STATUSES.has(r.status)) {
        const err = new Error(`HTTP ${r.status}`) as Error & { status?: number };
        err.status = r.status;
        throw err;
      }
      return r;
    }, { /* ... retryOn unchanged ... */ });
```

```ts
import nock from 'nock';
import { setGlobalDispatcher, MockAgent } from 'undici';

interface Cassette {
  request: { method: string; url: string };
  responses: Array<{ status: number; headers: Record<string, string>; body: unknown }>;
}

function loadCassette(name: string): Cassette {
  const fs = require('node:fs') as typeof import('node:fs');
  const path = require('node:path') as typeof import('node:path');
  return JSON.parse(fs.readFileSync(
    path.resolve('tests/fixtures/http-cassettes', `${name}.json`), 'utf8'));
}

function applyCassette(cassette: Cassette): MockAgent {
  const agent = new MockAgent();
  agent.disableNetConnect();
  setGlobalDispatcher(agent);
  const url = new URL(cassette.request.url);
  const pool = agent.get(url.origin);
  for (const r of cassette.responses) {
    pool.intercept({ path: url.pathname + url.search, method: cassette.request.method as 'GET' })
        .reply(r.status, r.body, { headers: r.headers });
  }
  return agent;
}
```
